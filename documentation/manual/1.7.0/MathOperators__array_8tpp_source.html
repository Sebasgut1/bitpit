753 template <class T, size_t d>
754 double norm(
755  const std::array<T, d> &x,
756  int p
757 ) {
758 
759 using namespace std;
760 
761 // ================================================================================== //
762 // VARIABLES DECLARATION //
763 // ================================================================================== //
764 
765 // Local variables
766 double z = 0.0;
767 double t, y;
768 
769 // Counters
770 size_t i;
771 int j;
772 
773 // ================================================================================== //
774 // COMPUTE THE P-NORM //
775 // ================================================================================== //
776 if (p == 1) { return(norm1(x)); }
777 if (p == 2) { return(norm2(x)); }
778 
779 if (d > 0) {
780  for (i = 0; i < d; i++) {
781  y = 1.0;
782  t = abs(x[i]);
783  for (j = 1; j <= p; j++) {
784  y = y*t;
785  } //next j
786  //z += pow(abs(x[i]), p);
787  z += y;
788  } //next i
789 }
790 
791 return(std::exp(std::log(std::max(z, 1.0e-307))/((double) p))); };

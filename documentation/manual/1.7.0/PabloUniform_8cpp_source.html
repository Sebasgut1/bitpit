682  void
683  PabloUniform::getBoundingBox(darray3 & P0, darray3 & P1) const {
684  // If there are no octants the bounding box is empty
685  uint32_t nocts = ParaTree::getNumOctants();
686  if (nocts == 0) {
687  P0 = getOrigin();
688  P1 = P0;
689 
690  return;
691  }
692 
693  // If the octree is serial we can evaluate the bounding box easily
694  // otherwise we need to scan all the octants
695  if (getSerial()) {
696  P0 = getOrigin();
697  P1 = P0;
698  for (int i=0; i<ParaTree::getDim(); i++){
699  P1[i] += getL();
700  }
701 
702  return;
703  }
704 
705  // If the octree is parallel we need to scan all the octants
706  darray3 cnode0, cnode1;
707 
708  uint32_t id = 0;
709  uint8_t nnodes = ParaTree::getNnodes();
710 
711  P0 = getNode(id, 0);
712  P1 = getNode(nocts-1, nnodes-1);
713 
714  for (id=0; id<nocts; id++){
715  cnode0 = getNode(id, 0);
716  cnode1 = getNode(id, nnodes-1);
717  for (int i=0; i<ParaTree::getDim(); i++){
718  P0[i] = min(P0[i], cnode0[i]);
719  P1[i] = max(P1[i], cnode1[i]);
720  }
721  }
722  };

873  void
874  PabloUniform::write(const std::string &filename) {
875 
876  if (getConnectivity().size() == 0) {
877  computeConnectivity();
878  }
879 
880  stringstream name;
881  name << "s" << std::setfill('0') << std::setw(4) << getNproc() << "-p" << std::setfill('0') << std::setw(4) << getRank() << "-" << filename << ".vtu";
882 
883  ofstream out(name.str().c_str());
884  if(!out.is_open()){
885  stringstream ss;
886  ss << filename << "*.vtu cannot be opened and it won't be written." << endl;
887  getLog() << ss.str();
888  return;
889  }
890  int nofNodes = getNumNodes();
891  int nofOctants = getNumOctants();
892  int nofGhosts = getNumGhosts();
893  int nofAll = nofGhosts + nofOctants;
894  out << "<?xml version=\"1.0\"?>" << endl
895  << "<VTKFile type=\"UnstructuredGrid\" version=\"0.1\" byte_order=\"BigEndian\">" << endl
896  << " <UnstructuredGrid>" << endl
897  << " <Piece NumberOfCells=\"" << getConnectivity().size() + getGhostConnectivity().size() << "\" NumberOfPoints=\"" << getNumNodes() << "\">" << endl;
898  out << " <Points>" << endl
899  << " <DataArray type=\"Float64\" Name=\"Coordinates\" NumberOfComponents=\""<< 3 <<"\" format=\"ascii\">" << endl
900  << " " << std::fixed;
901  for(int i = 0; i < nofNodes; i++)
902  {
903  const std::array<double,3> & nodeCoordinates = getNodeCoordinates(i);
904  for(int j = 0; j < 3; ++j){
905  out << std::setprecision(6) << nodeCoordinates[j] << " ";
906  }
907  if((i+1)%4==0 && i!=nofNodes-1)
908  out << endl << " ";
909  }
910  out << endl << " </DataArray>" << endl
911  << " </Points>" << endl
912  << " <Cells>" << endl
913  << " <DataArray type=\"UInt64\" Name=\"connectivity\" NumberOfComponents=\"1\" format=\"ascii\">" << endl
914  << " ";
915  for(int i = 0; i < nofOctants; i++)
916  {
917  for(int j = 0; j < getNnodes(); j++)
918  {
919  int jj = j;
920  if (getDim()==2){
921  if (j<2){
922  jj = j;
923  }
924  else if(j==2){
925  jj = 3;
926  }
927  else if(j==3){
928  jj = 2;
929  }
930  }
931  out << getConnectivity()[i][jj] << " ";
932  }
933  if((i+1)%3==0 && i!=nofOctants-1)
934  out << endl << " ";
935  }
936  for(int i = 0; i < nofGhosts; i++)
937  {
938  for(int j = 0; j < getNnodes(); j++)
939  {
940  int jj = j;
941  if (getDim()==2){
942  if (j<2){
943  jj = j;
944  }
945  else if(j==2){
946  jj = 3;
947  }
948  else if(j==3){
949  jj = 2;
950  }
951  }
952  out << getGhostConnectivity()[i][jj] << " ";
953  }
954  if((i+1)%3==0 && i!=nofGhosts-1)
955  out << endl << " ";
956  }
957  out << endl << " </DataArray>" << endl
958  << " <DataArray type=\"UInt64\" Name=\"offsets\" NumberOfComponents=\"1\" format=\"ascii\">" << endl
959  << " ";
960  for(int i = 0; i < nofAll; i++)
961  {
962  out << (i+1)*getNnodes() << " ";
963  if((i+1)%12==0 && i!=nofAll-1)
964  out << endl << " ";
965  }
966  out << endl << " </DataArray>" << endl
967  << " <DataArray type=\"UInt8\" Name=\"types\" NumberOfComponents=\"1\" format=\"ascii\">" << endl
968  << " ";
969  for(int i = 0; i < nofAll; i++)
970  {
971  int type;
972  type = 5 + (getDim()*2);
973  out << type << " ";
974  if((i+1)%12==0 && i!=nofAll-1)
975  out << endl << " ";
976  }
977  out << endl << " </DataArray>" << endl

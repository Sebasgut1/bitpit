670 void POD::setSensorMask(const PiercedStorage<bool> & mask, VolumeKernel * mesh)
671 {
672  m_sensorMask.setStaticKernel(&(m_podkernel->getMesh()->getCells()));
673 
674  if (m_staticMesh || mesh == m_podkernel->getMesh()){
675 
676  for (const Cell &cell : m_podkernel->getMesh()->getCells()) {
677  long id = cell.getId();
678  m_sensorMask[id] = mask[id];
679  }
680 
681  }
682  else{
683 
684  if (mesh == nullptr)
685  throw std::runtime_error ("POD: null mesh pointer passed to setSensorMask");
686 
687  //If mapping dirty or not computed, compute mapping of input mesh on pod mesh
688  if (m_podkernel->isMapperDirty())
689  _computeMapper(mesh);
690 
691  std::unordered_set<long> trueCells;
692  for (const Cell & cell : mesh->getCells()){
693  long id = cell.getId();
694  if (mask[id])
695  trueCells.insert(id);
696  }
697 
698  //Map true cells
699  std::unordered_set<long> mappedCells = m_podkernel->mapCellsToPOD(&trueCells);
700 
701  m_podkernel->mapBoolFieldToPOD(mask, mesh, &mappedCells, m_sensorMask);
702 
703  }
704 
705  fillListActiveIDs(m_sensorMask);
706  m_toUpdate = true;
707 
708 }

975 void POD::_evalMeanMesh()
976 {
977 
978  if (m_podkernel->getMesh() == nullptr){
979 
980  //Read first mesh and use as meshPOD
981  VolumeKernel* meshr = m_podkernel->readMesh(m_database[0]);
982  m_podkernel->setMesh(meshr);
983 
984  //Dynamic mesh case
985  if (!m_staticMesh){
986  // Compute meshPOD (starting from inital mesh) and fill filter
987  for (std::size_t i = 1; i < m_nSnapshots; ++i) {
988  log::cout() << "pod : evaluation POD mesh - use snapshot " << i+1 << "/" << m_nSnapshots << std::endl;
989  VolumeKernel* readmesh = m_podkernel->readMesh(m_database[i]);
990  m_podkernel->adaptMeshToMesh(m_podkernel->getMesh(), readmesh);
991  m_podkernel->clearMapper();
992  }
993  }
994  }
995 
996  {
997  // Read first snapshot to set n scalar and vector fields
998  pod::PODField readf;
999  readSnapshot(m_database[0], readf);
1000  }
1001 
1002  //Compute cells volume
1003  m_podkernel->evalCellsVolume();
1004 
1005  m_filter.setStaticKernel(&(m_podkernel->getMesh()->getCells()));
1006  m_filter.fill(true);
1007 
1008 
1009  if (m_useMean){
1010  m_mean.scalar = std::unique_ptr<pod::ScalarStorage>(new pod::ScalarStorage(m_nScalarFields, &m_podkernel->getMesh()->getCells()));
1011  m_mean.vector = std::unique_ptr<pod::VectorStorage>(new pod::VectorStorage(m_nVectorFields, &m_podkernel->getMesh()->getCells()));
1012  m_mean.scalar->fill(0.0);
1013  m_mean.vector->fill(std::array<double, 3>{{0.0, 0.0, 0.0}});
1014 
1015  // Compute mean (starting from first snapshot) & filter+mask
1016  for (std::size_t i = 0; i < m_nSnapshots; ++i) {
1017  log::cout() << "pod : evaluation mean - use snapshot " << i+1 << "/" << m_nSnapshots << std::endl;
1018  pod::PODField readf;
1019  readSnapshot(m_database[i], readf);
1020  if (!m_staticMesh){
1021  //Compute mapping of read mesh on pod mesh
1022  _computeMapper(readf.mesh);
1023  readf = pod::PODField(m_podkernel->mapPODFieldToPOD(readf, nullptr));
1024  m_podkernel->clearMapper();
1025  }
1026 
1027  for (const Cell &cell : m_podkernel->getMesh()->getCells()) {
1028  m_filter[cell.getId()] = (m_filter[cell.getId()]) && (readf.mask->at(cell.getId())); //N.B. mask == filter
1029  if (m_filter[cell.getId()]){
1030  for (std::size_t j = 0; j < m_nScalarFields; ++j)
1031  m_mean.scalar->at(cell.getId(),j) += readf.scalar->at(cell.getId(),j) / double(m_nSnapshots);
1032  for (std::size_t j = 0; j < m_nVectorFields; ++j)
1033  m_mean.vector->at(cell.getId(),j) += readf.vector->at(cell.getId(),j) / double(m_nSnapshots);
1034  }
1035  else{

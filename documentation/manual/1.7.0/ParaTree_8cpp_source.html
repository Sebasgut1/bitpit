274  ParaTree::_initializeCommunicator(MPI_Comm communicator)
275  {
276  // Communicator can be set just once
277  if (isCommSet()) {
278  throw std::runtime_error ("PABLO communicator can be set just once");
279  }
280 
281  // Early return if the communicator is a null communicator
282  if (communicator == MPI_COMM_NULL) {
283  m_comm = MPI_COMM_NULL;
284  return;
285  }

288  //
289  // No library routine should use MPI_COMM_WORLD as the communicator;

291  // be used.
292  MPI_Comm_dup(communicator, &m_comm);

300  void
301  ParaTree::_initializePartitions() {
302 #if BITPIT_ENABLE_MPI==1
303  // Set MPI information
304  if (isCommSet()) {
305  MPI_Comm_size(m_comm, &m_nproc);
306  MPI_Comm_rank(m_comm, &m_rank);
307  } else {
308  m_rank = 0;
309  m_nproc = 1;
310  }
311 #else
312  // Set dummy MPI information
313  m_rank = 0;
314  m_nproc = 1;
315 #endif
316 
317  // Create the data structures for storing partion information
318  m_partitionFirstDesc.resize(m_nproc);
319  m_partitionLastDesc.resize(m_nproc);
320  m_partitionRangeGlobalIdx.resize(m_nproc);
321  m_partitionRangeGlobalIdx0.resize(m_nproc);
322  uint64_t lastDescMorton = m_octree.getLastDescMorton();
323  uint64_t firstDescMorton = m_octree.getFirstDescMorton();
324  for(int p = 0; p < m_nproc; ++p){
325  m_partitionRangeGlobalIdx0[p] = 0;
326  m_partitionRangeGlobalIdx[p] = m_globalNumOctants - 1;
327  m_partitionLastDesc[p] = lastDescMorton;
328  m_partitionFirstDesc[p] = firstDescMorton;
329  }

493  void ParaTree::dump(std::ostream &stream, bool full)
494  {
495  // Version
496  utils::binary::write(stream, getDumpVersion());
497 
498  // Tree data
499  utils::binary::write(stream, getNproc());
500 
501  utils::binary::write(stream, getDim());
502 
503  utils::binary::write(stream, getSerial());
504  utils::binary::write(stream, getNofGhostLayers());
505  utils::binary::write(stream, getMaxDepth());
506  utils::binary::write(stream, getStatus());
507  utils::binary::write(stream, getBalanceCodimension());
508 
509  for (int i = 0; i < m_treeConstants->nFaces; i++) {
510  utils::binary::write(stream, getPeriodic(i));
511  }
512 
513  // Octant data
514  uint32_t nOctants = getNumOctants();
515  utils::binary::write(stream, nOctants);
516 
517  uint32_t nGlobalOctants = getGlobalNumOctants();
518  utils::binary::write(stream, nGlobalOctants);
519 
520  for (uint32_t i = 0; i < nOctants; i++) {
521  const Octant &octant = m_octree.m_octants[i];
522 
523  utils::binary::write(stream, octant.getLevel());
524  utils::binary::write(stream, octant.getLogicalX());
525  utils::binary::write(stream, octant.getLogicalY());
526  utils::binary::write(stream, octant.getLogicalZ());
527  utils::binary::write(stream, octant.getGhostLayer());
528 
529  for (size_t k = 0; k < Octant::INFO_ITEM_COUNT; ++k) {
530  utils::binary::write(stream, (bool) octant.m_info[k]);
531  }
532 
533  utils::binary::write(stream, octant.getBalance());
534  utils::binary::write(stream, octant.getMarker());
535  }
536 
537  // Information about partitioning
538  for (int k = 0; k < m_nproc; ++k) {
539  utils::binary::write(stream, m_partitionFirstDesc[k]);
540  }
541 
542  for (int k = 0; k < m_nproc; ++k) {
543  utils::binary::write(stream, m_partitionLastDesc[k]);
544  }
545 
546  for (int k = 0; k < m_nproc; ++k) {
547  utils::binary::write(stream, m_partitionRangeGlobalIdx[k]);
548  }
549 
550  // Extended information (mapping, ...)
551  utils::binary::write(stream, full);
552  if (full) {
553  utils::binary::write(stream, m_lastOp);
554  if (m_lastOp == OP_ADAPT_MAPPED){
555  for (auto idx : m_mapIdx) {
556  utils::binary::write(stream, idx);
557  }
558 
559  utils::binary::write(stream, m_octree.m_lastGhostBros.size());
560  for (auto lastGhostBrother : m_octree.m_lastGhostBros) {
561  utils::binary::write(stream, lastGhostBrother);
562  }
563  }
564  else if (m_lastOp == OP_LOADBALANCE || m_lastOp == OP_LOADBALANCE_FIRST){
565  for (int i = 0; i < m_nproc; ++i) {
566  utils::binary::write(stream, m_partitionRangeGlobalIdx0[i]);
567  }
568  }
569  }
570 
571  }

579  void ParaTree::restore(std::istream &stream)
580  {
581  // Version
582  int version;
583  utils::binary::read(stream, version);
584  if (version != getDumpVersion()) {
585  throw std::runtime_error ("The version of the file does not match the required version");
586  }
587 
588  // Check if the number of processes matches
589  int nProcs;
590  utils::binary::read(stream, nProcs);
591  if (nProcs != m_nproc) {
592  throw std::runtime_error ("The restart was saved with a different number of processes.");
593  }
594 
595  // Initialize the tree
596  uint8_t dimension;
597  utils::binary::read(stream, dimension);
598 
599  m_octree.initialize(dimension);
600  m_trans.initialize(dimension);
601  reinitialize(dimension, m_log->getName());
602  reset(false);
603 
604  // Set tree properties
605  utils::binary::read(stream, m_serial);
606  utils::binary::read(stream, m_nofGhostLayers);
607  utils::binary::read(stream, m_maxDepth);
608  utils::binary::read(stream, m_status);
609 
610  bool balanceCodimension;
611  utils::binary::read(stream, balanceCodimension);
612  setBalanceCodimension(balanceCodimension);
613 
614  for (int i = 0; i < m_treeConstants->nFaces; i++) {
615  bool periodicBorder;
616  utils::binary::read(stream, periodicBorder);
617  if (periodicBorder){
618  setPeriodic(i);
619  }
620  }
621 
622  // Restore octants
623  uint32_t nOctants;
624  utils::binary::read(stream, nOctants);
625  m_octree.m_sizeOctants = nOctants;
626 
627  uint32_t nGlobalOctants;
628  utils::binary::read(stream, nGlobalOctants);
629  m_globalNumOctants = nGlobalOctants;
630 
631  m_octree.m_octants.clear();
632  m_octree.m_octants.reserve(nOctants);
633  for (uint32_t i = 0; i < nOctants; i++) {
634  // Create octant
635  uint8_t level;
636  utils::binary::read(stream, level);
637 
638  uint32_t x;
639  utils::binary::read(stream, x);
640 
641  uint32_t y;
642  utils::binary::read(stream, y);
643 
644  uint32_t z;
645  utils::binary::read(stream, z);
646 
647  Octant octant(false, m_dim, level, x, y, z);
648 
649  int ghost;
650  utils::binary::read(stream, ghost);
651  octant.setGhostLayer(ghost);
652 
653  // Set octant info
654  for (size_t k = 0; k < Octant::INFO_ITEM_COUNT; ++k) {
655  bool bit;
656  utils::binary::read(stream, bit);
657  octant.m_info.set(k, bit);
658  }
659 
660  // Set octant 2:1 balance
661  bool balance21;
662  utils::binary::read(stream, balance21);
663  octant.setBalance(balance21);
664 
665  // Set marker
666  int8_t marker;
667  utils::binary::read(stream, marker);
668  octant.setMarker(marker);
669 
670  // Add octant to the list
671  m_octree.m_octants.push_back(std::move(octant));
672  }
673 
674  m_octree.updateLocalMaxDepth();
675 
676  // Set first last descendant
677  m_partitionFirstDesc.resize(m_nproc);
678  for (int k = 0; k < m_nproc; ++k) {
679  uint64_t descendant;
680  utils::binary::read(stream, descendant);
681  m_partitionFirstDesc[k] = descendant;
682  }
683  m_octree.m_firstDescMorton = m_partitionFirstDesc[m_rank];
684 
685  m_partitionLastDesc.resize(m_nproc);
686  for (int k = 0; k < m_nproc; ++k) {
687  uint64_t descendant;
688  utils::binary::read(stream, descendant);
689  m_partitionLastDesc[k] = descendant;
690  }
691  m_octree.m_lastDescMorton = m_partitionLastDesc[m_rank];
692 
693  // Set partitions and parallel information
694  m_partitionRangeGlobalIdx.resize(m_nproc);
695  for (int k = 0; k < m_nproc; ++k) {
696  uint64_t rangeGlobalIdx;
697  utils::binary::read(stream, rangeGlobalIdx);
698  m_partitionRangeGlobalIdx[k] = rangeGlobalIdx;
699  }
700 
701 #if BITPIT_ENABLE_MPI==1
702  if (!m_serial) {
703  computeGhostHalo();
704  }
705 #endif
706 
707  // Full restore (i.e. restore with mapper of last operation)
708  m_mapIdx.clear();
709 
710  for (int i = 0; i < m_nproc; ++i) {
711  m_partitionRangeGlobalIdx0[i] = 0;
712  }
713 
714  bool full;
715  utils::binary::read(stream, full);
716  if (full) {
717  utils::binary::read(stream, m_lastOp);
718  if (m_lastOp == OP_ADAPT_MAPPED) {
719  m_mapIdx.resize(m_octree.m_octants.size());
720  for (size_t i = 0; i < m_octree.m_octants.size(); ++i) {
721  utils::binary::read(stream, m_mapIdx[i]);
722  }
723 
724  size_t lastGhostBrosSize;
725  utils::binary::read(stream, lastGhostBrosSize);
726  m_octree.m_lastGhostBros.resize(lastGhostBrosSize);
727  for (size_t i = 0; i < lastGhostBrosSize; ++i) {
728  utils::binary::read(stream, m_octree.m_lastGhostBros[i]);
729  }
730  }
731  else if (m_lastOp == OP_LOADBALANCE || m_lastOp == OP_LOADBALANCE_FIRST){
732  for (int i = 0; i < m_nproc; ++i) {
733  utils::binary::read(stream, m_partitionRangeGlobalIdx0[i]);
734  }
735  }
736  }
737  else {
738  m_lastOp = OP_INIT;
739  }
740  }

879  void
880  ParaTree::replaceComm(MPI_Comm communicator, MPI_Comm *previousCommunicator)
881  {
882  // The communicator has to be already set
883  if (!isCommSet()) {
884  throw std::runtime_error ("PABLO communicator is not set");
885  }
886 
887  // Check if the communicator is valid
888  //
889  // The communicator should be equaivalent to the one currently set,
890  // i.e., the rank of the processes have to be the same in both
891  // communicators.
892  int updatedRank;
893  MPI_Comm_rank(communicator, &updatedRank);
894  int currentRank = getRank();
895 
896  int isValid = (updatedRank == currentRank) ? 1 : 0;

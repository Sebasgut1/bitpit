385 void LevelSet::addProcessingOrder( int objectId ) {
386 
387  bool primary = m_objects.at(objectId)->isPrimary() ;
388 
389  if(primary){
390  std::vector<int>::iterator orderItr = m_order.begin() ;
391  bool iterate( orderItr != m_order.end()) ;
392 
393  while(iterate){
394  int id = *orderItr ;
395  if( m_objects.at(id)->isPrimary() ){
396  ++orderItr ;
397  iterate = orderItr != m_order.end() ;
398  } else {
399  iterate = false;
400  }
401  }
402 
403  m_order.insert(orderItr,objectId) ;
404 
405  } else {
406  m_order.push_back(objectId) ;
407 
408  }
409 
410 }

423  if(*orderItr==objectId){

429  return false ;

560 void LevelSet::update( const std::vector<adaption::Info> &mapper ){
561 
562  assert(m_kernel && "LevelSet::setMesh() must be called prior to LevelSet::update()");
563 
564  // Udate the cache of the kernel
565  m_kernel->updateGeometryCache( mapper ) ;
566 
567  // Update ls in narrow band
568  for( int objectId : m_order){
569  auto &visitor = *(m_objects.at(objectId)) ;
570  visitor.clearAfterMeshAdaption(mapper) ;
571  visitor.updateLSInNarrowBand( mapper, m_signedDF ) ;
572 #if BITPIT_ENABLE_MPI
573  visitor.exchangeGhosts();
574 #endif
575  if(m_propagateS) visitor.propagateSign() ;
576  }

584 void LevelSet::partition( const std::vector<adaption::Info> &mapper ){
585 
586  assert(m_kernel && "LevelSet::setMesh() must be called prior to LevelSet::partition()");
587 
588  // Set the communicator
589  if (!m_kernel->isCommunicatorSet()) {
590  m_kernel->initializeCommunicator();
591  }
592 
593  // Udate the cache of the kernel
594  m_kernel->updateGeometryCache( mapper ) ;
595 
596  // Compile send and receive lists
597  std::unordered_map<int,std::vector<long>> sendList, recvList ;
598  for( const auto &event : mapper){
599  if( event.entity != adaption::Entity::ENTITY_CELL){
600  continue;
601  }
602 
603  if( event.type == adaption::Type::TYPE_PARTITION_SEND){
604  sendList.insert({{event.rank,event.previous}}) ;
605  } else if( event.type == adaption::Type::TYPE_PARTITION_RECV){
606  recvList.insert({{event.rank,event.current}}) ;
607  }
608  }
609 
610  // Communicate according to new partitioning
611  for( int objectId : m_order){
612  auto &visitor = *(m_objects.at(objectId)) ;
613 
614  visitor.communicate( sendList, recvList, &mapper ) ;
615  }
616 
617  for( int objectId : m_order){
618  auto &visitor = *(m_objects.at(objectId)) ;
619 
620  visitor.exchangeGhosts();
621  }
622 
623 }

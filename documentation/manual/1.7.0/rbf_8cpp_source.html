573 std::vector<double> RBFKernel::evalRBF( const std::array<double,3> &point)
574 {
575  std::vector<double> values(m_fields, 0.);
576  int i, j;
577  double dist, basis;
578 
579  for( i=0; i<m_nodes; ++i ){
580  if( m_activeNodes[i] ) {
581  dist = calcDist(point, i) / m_supportRadius;
582  basis = evalBasis( dist );
583 
584  for( j=0; j<m_fields; ++j) {
585  values[j] += basis * m_weight[j][i];
586  }
587  }
588  }
589 
590  return values;
591 }

632 int RBFKernel::solve()
633 {
634  if(m_mode == RBFMode::PARAM) {
635  return -1;
636  }
637 
638  int j, k;
639  double dist;
640 
641  int nS = getActiveCount();
642  int nrhs = getDataCount();
643 
644  int lda = nS;
645  int ldb = nS;
646  int info;
647  int ipiv[nS];
648 
649  std::vector<int> activeSet( getActiveSet() );
650 
651  std::vector<double> A(lda * nS);
652  std::vector<double> b(ldb * nrhs);
653 
654  k=0;
655  for( j=0; j<nrhs; ++j) {
656  for( const auto & i : activeSet ) {
657  b[k] = m_value[j][i];
658  ++k;
659  }
660  }
661 
662  k=0;
663  for( const auto &i : activeSet ) {
664  for( const auto &j : activeSet ){
665 
666  dist = calcDist(j,i) / m_supportRadius;
667  A[k] = evalBasis( dist );
668  k++;
669  }
670  }
671 
672  info = LAPACKE_dgesv( LAPACK_COL_MAJOR, nS, nrhs, A.data(), lda, ipiv, b.data(), ldb );
673 
674  if( info > 0 ) {
675  printf( "The diagonal element of the triangular factor of the linear system matrix \n" );
676  printf( "U(%i,%i) is zero, so that matrix is singular;\n", info, info );
677  printf( "the solution could not be computed.\n" );
678  return 1;
679  }
680 
681  m_weight.resize(nrhs);
682 
683  k=0;
684  for( j=0; j<nrhs; ++j) {
685  m_weight[j].resize(m_nodes,0);
686 
687  for( const auto &i : activeSet ) {
688  m_weight[j][i] = b[k];
689  ++k;
690  }
691  }
692 
693  return 0;
694 }

719  local[j] = m_value[j][i];

721

723 
724  }

746  }

836 double RBFKernel::evalError( )
837 {
838  if(m_mode == RBFMode::PARAM) {
839  return -1.0;
840  }
841 
842  int i(0), j(0);
843  //int index;
844  double maxError(0), relError, realValue, norm;
845  std::vector<double> reconValues;
846 
847  for( int iX=0; iX< m_nodes; ++iX ) {
848  reconValues = evalRBF( iX );
849 
850  j=0;
851  norm = 0.;
852  relError = 0.;
853  for( auto &val : reconValues ) {
854  realValue = m_value[j][i];
855  relError += std::pow( (val - realValue), 2 );
856  norm += std::pow( realValue, 2 );
857 
858  ++j;
859  }
860 
861  relError = sqrt(relError);
862  m_error[i] = relError;
863 
864  if( relError > maxError ) {
865  maxError = relError;
866  }
867 
868  ++i;
869  }
870 
871  return maxError;
872 }

880 int RBFKernel::solveLSQ()
881 {
882  if(m_mode == RBFMode::PARAM) {
883  return -1;
884  }
885 
886  int i, j, k;
887  double dist;
888 
889  int nR = getActiveCount();
890  int nP = m_nodes;
891  int nrhs = getDataCount();
892 
893  std::vector<int> activeSet( getActiveSet() );
894 
895  int n ,m, lda, ldb, info, rank;
896  double rcond = -1.0;
897  //double rcond = 1.e-4;
898 
899  m = nP;
900  n = nR;
901 
902  lda = m;
903  ldb = std::max(n,m);
904 
905  std::vector<double> A(lda * n);
906  std::vector<double> b(ldb * nrhs);
907  std::vector<double> s(m);
908 
909  for( j=0; j<nrhs; ++j) {
910  for( i=0; i<nP; ++i) {
911  k = j*ldb + i;
912  b[k] = m_value[j][i];
913  }
914  }
915 
916  k=0;
917  for( const auto &j : activeSet ) {
918  for( i=0; i<nP; ++i) {
919  dist = calcDist(j,i) / m_supportRadius;
920  A[k] = evalBasis( dist );
921  k++;
922  }
923  }
924 
925  info = LAPACKE_dgelsd( LAPACK_COL_MAJOR, nP, nR, nrhs, A.data(), lda, b.data(), ldb, s.data(), rcond, &rank );
926 
927  if( info > 0 ) {
928  return 1;
929  }
930 
931  m_weight.resize(nrhs);
932 
933  for( j=0; j<nrhs; ++j) {
934  m_weight[j].clear();
935  m_weight[j].resize(m_nodes,0);
936 
937  k=0;
938  for( const auto &i : activeSet ) {
939  m_weight[j][i] = b[j*ldb+k];
940  ++k;
941  }
942  }
943 
944  return 0;
945 }

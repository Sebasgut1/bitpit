70 void SkdPatchInfo::buildCache(const PatchKernel::CellConstRange &cellRange)
71 {
72  m_cellBoxes = std::unique_ptr<BoxCache>(new BoxCache(2, &(m_patch->getCells())));
73  m_cellCentroids = std::unique_ptr<BoxCache>(new CentroidCache(1, &(m_patch->getCells())));
74  for (auto itr = cellRange.cbegin(); itr != cellRange.cend(); ++itr) {
75  std::size_t rawCellId = itr.getRawIndex();
76 
77  // Cell info
78  const Cell &cell = *itr;
79  int nCellVertices = cell.getVertexCount();
80  ConstProxyVector<long> cellConnect = cell.getVertexIds();
81 
82  // Bounding box
83  std::array<double, 3> &cellBoxMin = m_cellBoxes->rawAt(rawCellId, 0);
84  std::array<double, 3> &cellBoxMax = m_cellBoxes->rawAt(rawCellId, 1);
85  std::array<double, 3> &cellCentroid = m_cellCentroids->rawAt(rawCellId, 0);
86 
87  cellBoxMin = m_patch->getVertexCoords(cellConnect[0]);
88  cellBoxMax = cellBoxMin;
89  cellCentroid = cellBoxMin;
90  for (int i = 1; i < nCellVertices; ++i) {
91  const std::array<double, 3> &coords = m_patch->getVertexCoords(cellConnect[i]);
92  for (int d = 0; d < 3; ++d) {
93  cellBoxMin[d] = std::min(coords[d], cellBoxMin[d]);
94  cellBoxMax[d] = std::max(coords[d], cellBoxMax[d]);
95  cellCentroid[d] += coords[d];
96  }
97  }
98  cellCentroid /= double(nCellVertices);
99  }
100 }

299 bool SkdBox::boxIntersectsSphere(const std::array<double, 3> &center, double radius) const
300 {
301  // If the box contains the center it will also intersect the sphere
302  if (boxContainsPoint(center, 0.)) {
303  return true;
304  }
305 
306  // Check if the distance between the center and its projection on
307  // the bounding box is smaller than the radius of the sphere.
308  std::array<double, 3> delta;
309  for (int d = 0; d < 3; d++) {
310  delta[d] = std::min(std::max(center[d], m_boxMin[d]), m_boxMax[d]) - center[d];
311  }
312  double distance = dotProduct(delta, delta);
313 
314  return (distance < radius * radius);
315 }

358 void SkdNode::initializeBoundingBox()
359 {
360  // Early return if there are no cells
361  if (m_cellRangeBegin == m_cellRangeEnd){
362  m_boxMin.fill( std::numeric_limits<double>::max());
363  m_boxMax.fill(- std::numeric_limits<double>::max());
364 
365  return;
366  }
367 
368  // Evaluate the bounding box
369  const std::vector<std::size_t> &cellRawIds = m_patchInfo->getCellRawIds();
370 
371  m_boxMin = m_patchInfo->getCachedBoxMin(cellRawIds[m_cellRangeBegin]);
372  m_boxMax = m_patchInfo->getCachedBoxMax(cellRawIds[m_cellRangeBegin]);
373  for (std::size_t n = m_cellRangeBegin + 1; n < m_cellRangeEnd; n++) {
374  const std::size_t rawCellId = cellRawIds[n];
375  const std::array<double, 3> &cellBoxMin = m_patchInfo->getCachedBoxMin(rawCellId);
376  const std::array<double, 3> &cellBoxMax = m_patchInfo->getCachedBoxMax(rawCellId);
377  for (int d = 0; d < 3; ++d) {
378  m_boxMin[d] = std::min(cellBoxMin[d], m_boxMin[d]);
379  m_boxMax[d] = std::max(cellBoxMax[d], m_boxMax[d]);
380  }
381  }
382 
383  // Inlfate the bounding box by a small epsilon
384  //
385  // The small inflation allows us to test intersection of a query
386  // against a safety box and be assured that non-intersection of
387  // with the box implies non-intersection with all the contents
388  // in the box, even when computations are susceptible to round-off
389  // error.
390  double tolerance = m_patchInfo->getPatch().getTol();
391 
392  m_boxMin -= tolerance;
393  m_boxMax += tolerance;
394 }

577 void SkdNode::updatePointClosestCell(const std::array<double, 3> &point, bool interiorCellsOnly,
578  long *id, double *distance) const
579 {
580  if (getCellCount() == 0) {
581  return;
582  }
583 
584  const PatchKernel &patch = m_patchInfo->getPatch();
585  const PiercedVector<Cell> &cells = patch.getCells();
586  const std::vector<std::size_t> &cellRawIds = m_patchInfo->getCellRawIds();
587 
588  std::array<std::array<double, 3>, ReferenceElementInfo::MAX_ELEM_VERTICES> staticCellVertexCoordinates;
589  std::vector<std::array<double, 3>> dynamicCellVertexCoordinates;
590 
591  for (std::size_t n = m_cellRangeBegin; n < m_cellRangeEnd; n++) {
592  std::size_t cellRawId = cellRawIds[n];
593  const Cell &cell = cells.rawAt(cellRawId);
594  if (interiorCellsOnly && !cell.isInterior()) {
595  continue;
596  }
597 
598  // Get the vertices ids
599  ConstProxyVector<long> elementVertexIds = cell.getVertexIds();
600  const int nElementVertices = elementVertexIds.size();
601 
602  // Get vertex coordinates
603  std::array<double, 3> *cellVertexCoordinates;
604  if (nElementVertices <= ReferenceElementInfo::MAX_ELEM_VERTICES) {
605  cellVertexCoordinates = staticCellVertexCoordinates.data();
606  } else {
607  dynamicCellVertexCoordinates.resize(nElementVertices);
608  cellVertexCoordinates = dynamicCellVertexCoordinates.data();
609  }
610 
611  for (int i = 0; i < nElementVertices; ++i) {
612  cellVertexCoordinates[i] = patch.getVertex(elementVertexIds[i]).getCoords();
613  }
614 
615  // Evaluate the distance from the cell
616  double cellDistance = cell.evalPointDistance(point, cellVertexCoordinates);
617 
618  // Update closest distance
619  updateClosestCellInfo(point, cell.getId(), cellDistance, id, distance);
620  }
621 }

640 void SkdNode::updateClosestCellInfo(const std::array<double, 3> &point,
641  long cellId, double cellDistance,
642  long *closestId, double *closestDistance) const
643 {
644  // Detect if the specified cell is closer than the current closest cell
645  const int DISTANCE_CLOSER = - 1;
646  const int DISTANCE_EQUAL = 0;
647  const int DISTANCE_FARTHER = 1;
648 
649  int distanceFlag = DISTANCE_FARTHER;
650  if (utils::DoubleFloatingEqual()(cellDistance, *closestDistance)) {
651  distanceFlag = DISTANCE_EQUAL;
652  } else if (cellDistance < *closestDistance) {
653  distanceFlag = DISTANCE_CLOSER;
654  }
655 
656  // Consider the case where no closest cell is defined
657  //
658  // Even if the id of the closest cell is null, we may have an
659  // estimated of the closest cell distance. We need to update
660  // the closest cell information only if the current cells is
661  // closer than the estimate.
662  if (*closestId == Cell::NULL_ID) {
663  if (distanceFlag != DISTANCE_FARTHER) {
664  *closestId = cellId;
665  *closestDistance = cellDistance;
666  }
667 
668  return;
669  }
670 
671  // Update closest cell information accordingly to the distance flag:
672  // - if the specified cell is closer the specified cell will become the
673  // closest cell
674  // - if the two cells have the same distance, the closest cell will be
675  // chosen using the normal of the cells.
676  // - if the specified cell is farther there is nothing to do;
677  switch (distanceFlag) {
678 
679  case DISTANCE_CLOSER:
680  {
681  *closestId = cellId;
682  *closestDistance = cellDistance;
683 
684  break;
685  }
686 
687  case DISTANCE_EQUAL:
688  {
689  const PatchKernel &patch = m_patchInfo->getPatch();
690 
691  // Project point ont the specified cell
692  const Cell &cell = patch.getCell(cellId);
693 
694  ConstProxyVector<long> cellVertexIds = cell.getVertexIds();
695  const int nCellVertices = cellVertexIds.size();
696 
697  std::vector<std::array<double, 3>> cellVertexCoordinates(nCellVertices);
698  for (int i = 0; i < nCellVertices; ++i) {
699  cellVertexCoordinates[i] = patch.getVertex(cellVertexIds[i]).getCoords();
700  }
701 
702  double cellProjectionDistance;
703  std::array<double, 3> cellProjection;
704  cell.evalPointProjection(point, cellVertexCoordinates.data(), &cellProjection, &cellProjectionDistance);
705 
706  // Project point ont the closest cell
707  const Cell &closest = patch.getCell(*closestId);
708 
709  ConstProxyVector<long> closestVertexIds = closest.getVertexIds();
710  const int nClosestVertices = closestVertexIds.size();
711 
712  std::vector<std::array<double, 3>> closestVertexCoordinates(nClosestVertices);
713  for (int i = 0; i < nClosestVertices; ++i) {
714  closestVertexCoordinates[i] = patch.getVertex(closestVertexIds[i]).getCoords();
715  }
716 
717  double closestProjectionDistance;
718  std::array<double, 3> closestProjection;
719  closest.evalPointProjection(point, closestVertexCoordinates.data(), &closestProjection, &closestProjectionDistance);
720 
721  // Find normal of the cells at the centroids
722  //
723  // To be more precise, the normals should beevaluated on projection
724  // points.
725  std::array<double, 3> cellNormal = cell.evalNormal(cellVertexCoordinates.data());
726  std::array<double, 3> closestNormal = closest.evalNormal(closestVertexCoordinates.data());
727 
728  // Find cell more aligned with tespect to the normal
729  double cellAligement = std::abs(dotProduct(cellNormal, point - cellProjection));
730  double closestAligement = std::abs(dotProduct(closestNormal, point - closestProjection));
731 
732  // If the specified cell is more aligned than the closest cell, update
733  // the closest info.
734  if (cellAligement > closestAligement) {
735  *closestId = cellId;
736  *closestDistance = cellDistance;
737  }
738 
739  break;
740  }
741 
742  default:
743  {
744  // Nothing to do
745  }
746 
747  }
748 }

806 void SkdGlobalCellDistance::executeMPIMinOperation(SkdGlobalCellDistance *in, SkdGlobalCellDistance *inout, int *len, MPI_Datatype *datatype)
807 {
808  BITPIT_UNUSED(datatype);
809 
810  int nDistances = *len;
811  for (int i = 0; i < nDistances; ++i) {
812  bool updateDistance = false;
813  if (in[i].m_id != Cell::NULL_ID) {
814  if (inout[i].m_id == Cell::NULL_ID) {
815  updateDistance = true;
816  } else if (std::abs(in[i].m_distance) < std::abs(inout[i].m_distance)) {
817  updateDistance = true;
818  }
819  }
820 
821  if (updateDistance) {
822  inout[i] = in[i];
823  }
824  }

984 void PatchSkdTree::build(std::size_t leafThreshold, bool squeezeStorage)
985 {
986  const PatchKernel &patch = m_patchInfo.getPatch();
987 
988  // Clear existing tree
989  clear();
990 
991  // Initialize list of cell raw ids
992  std::size_t nCells;
993  PatchKernel::CellConstRange cellRange;
994  if (m_interiorCellsOnly) {
995  nCells = patch.getInternalCellCount();
996  cellRange.initialize(patch.internalCellConstBegin(), patch.internalCellConstEnd());
997  } else {
998  nCells = patch.getCellCount();
999  cellRange.initialize(patch.cellConstBegin(), patch.cellConstEnd());
1000  }
1001 
1002  if (m_cellRawIds.size() != nCells) {
1003  m_cellRawIds.resize(nCells);
1004  }
1005 
1006  std::size_t k = 0;
1007  for (auto itr = cellRange.begin(); itr != cellRange.end(); ++itr) {
1008  m_cellRawIds[k++] = itr.getRawIndex();
1009  }
1010 
1011  // Build patch cache
1012  m_patchInfo.buildCache(cellRange);
1013 
1014  // Initialize node list
1015  std::size_t nodesCount = std::max(1, int(std::ceil(2. * nCells / leafThreshold - 1.)));
1016  m_nodes.reserve(nodesCount);
1017 
1018  // Create the root
1019  m_nodes.emplace_back(&m_patchInfo, 0, nCells);
1020 
1021  // Create the tree
1022  std::queue<std::size_t> nodeStack;
1023  nodeStack.push(0);
1024  while (!nodeStack.empty()) {

1043  }

1181 void PatchSkdTree::createChildren(std::size_t parentId, std::size_t leafThreshold)
1182 {
1183  const SkdNode &parent = getNode(parentId);
1184 
1185  // Check if the parent is a leaf
1186  std::size_t parentCellCount = parent.getCellCount();
1187  if (parentCellCount <= leafThreshold) {
1188  createLeaf(parentId);
1189  return;
1190  }
1191 
1192  // Evaluate the preferred direction along which elements will be split.
1193  //
1194  // The elements will be split along a plane normal to the direction
1195  // for which the bounding box has the maximum length.
1196  const std::array<double, 3> &parentBoxMin = parent.getBoxMin();
1197  const std::array<double, 3> &parentBoxMax = parent.getBoxMax();
1198 
1199  int largerDirection = 0;
1200  double boxMaximumLength = parentBoxMax[largerDirection] - parentBoxMin[largerDirection];
1201  for (int d = 1; d < 3; ++d) {
1202  double length = parentBoxMax[d] - parentBoxMin[d];
1203  if (length > boxMaximumLength) {
1204  largerDirection = d;
1205  boxMaximumLength = length;
1206  }
1207  }
1208 
1209  // Split the elements.
1210  //
1211  // Preferred direction is tried first, if the split along this direction
1212  // will produce and empty child (e.g., all the cell's box centroids have
1213  // the same coordinate along the split direction and therefore all the
1214  // cells would be assigned to the left chiled) the split will be performed
1215  // along one of the other directions.
1216  std::array<double, 3> parentWeightedCentroid = parent.evalBoxWeightedMean();
1217  for (int d = 0; d < 3; ++d) {
1218  // Update the split direction
1219  int splitDirection = (largerDirection + d) % 3;
1220 
1221  // Get the threshold for the split
1222  double splitThreshold = parentWeightedCentroid[splitDirection];
1223 
1224  // Order the elements
1225  //
1226  // All the elements with a centroid coordinate less or equal than the
1227  // threshold will be assigned to the left child, the others will be
1228  // assigned to the right child.
1229  std::size_t leftBegin = parent.m_cellRangeBegin;
1230  std::size_t leftEnd = parent.m_cellRangeEnd;
1231  std::size_t rightBegin = parent.m_cellRangeBegin;
1232  std::size_t rightEnd = parent.m_cellRangeEnd;
1233  while (true) {
1234  // Update the right begin
1235  while (rightBegin != leftEnd && m_patchInfo.getCachedCentroid(m_cellRawIds[rightBegin])[splitDirection] <= splitThreshold) {
1236  rightBegin++;
1237  }
1238 
1239  // Update the left end
1240  while (rightBegin != leftEnd && m_patchInfo.getCachedCentroid(m_cellRawIds[leftEnd - 1])[splitDirection] > splitThreshold) {
1241  leftEnd--;
1242  }
1243 
1244  // If all the elements are in the right position we can exit
1245  if (rightBegin == leftEnd) {
1246  break;
1247  }
1248 
1249  // If left end and and right begin are not equal, that the two ids
1250  // point to misplaced elements. Swap the elements, advance the ids
1251  // and continue iterating.
1252  std::iter_swap(m_cellRawIds.begin() + (leftEnd - 1), m_cellRawIds.begin() + rightBegin);
1253  }
1254 
1255  if (leftEnd <= leftBegin || rightEnd <= rightBegin) {
1256  continue;
1257  }
1258 
1259  // Create the left child
1260  //
1261  // Adding new nodes may invalidate any pointer and reference to the
1262  // nodes, we cannot store a reference to the parent node.
1263  long leftId = m_nodes.size();
1264  m_nodes.emplace_back(&m_patchInfo, leftBegin, leftEnd);
1265  _getNode(parentId).m_children[static_cast<std::size_t>(SkdNode::CHILD_LEFT)] = leftId;
1266 
1267  // Create the right child
1268  //
1269  // Adding new nodes may invalidate any pointer and reference to the
1270  // nodes, we cannot store a reference to the parent node.
1271  long rightId = m_nodes.size();
1272  m_nodes.emplace_back(&m_patchInfo, rightBegin, rightEnd);
1273  _getNode(parentId).m_children[static_cast<std::size_t>(SkdNode::CHILD_RIGHT)] = rightId;
1274 
1275  // Done
1276  return;
1277  }
1278 
1279  // It was not possible to split the elements. They have all the same
1280  // characteristic position and therefore they will be clustered together
1281  // in a leaf node.
1282  createLeaf(parentId);
1283 }

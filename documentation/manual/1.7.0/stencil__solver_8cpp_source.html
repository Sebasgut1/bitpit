103 template<>
104 void DiscretizationStencilSolverAssembler<StencilBlock>::initializeBlockSize()
105 {
106  assert(m_blockSize == -1);
107 
108  std::size_t nStencils = m_stencils->size();
109 
110  // Set the block size equal to the size of the first weight
111  m_blockSize = - 1;
112  for (std::size_t i = 0; i < nStencils; ++i) {
113  const StencilBlock &stencil = (*m_stencils)[i];
114  std::size_t stencilSize = stencil.size();
115  if (stencilSize == 0) {
116  continue;
117  }
118 
119  const StencilBlock::weight_type *weightData = stencil.weightData();
120  m_blockSize = weightData[0].size();
121  break;
122  }
123 
124  assert(m_blockSize != -1);
125 
126 #ifdef DEBUG
127  // All weight sizes should match
128  for (std::size_t i = 0; i < nStencils; ++i) {
129  const StencilBlock &stencil = (*m_stencils)[i];
130  const StencilBlock::weight_type *weightData = stencil.weightData();
131  std::size_t stencilSize = stencil.size();
132 
133  for (std::size_t k = 0; k < stencilSize; ++k) {
134  assert((int) weightData[k].size() == m_blockSize);
135  }
136 
137  assert((int) stencil.getConstant().size() == m_blockSize);
138  }
139 #endif
140 }

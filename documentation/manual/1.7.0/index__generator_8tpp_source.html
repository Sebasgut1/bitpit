51 template<typename id_t>
52 typename IndexGenerator<id_t>::id_type IndexGenerator<id_t>::generate()
53 {
54  // If the trash is empty generate a new id otherwise recycle the first id
55  // in the trash.
56  if (m_trash.empty()) {
57  if (m_lowest == NULL_ID) {
58  m_lowest = 0;
59  m_highest = 0;
60  m_latest = m_lowest;
61  } else if (m_lowest > 0) {
62  --m_lowest;
63  m_latest = m_lowest;
64  } else {
65  assert(m_highest < std::numeric_limits<id_type>::max());
66  ++m_highest;
67  m_latest = m_highest;
68  }
69  } else {
70  m_latest = *(m_trash.begin());
71  eraseFromTrash(m_latest);
72  }
73 
74  return m_latest;
75 }

104 template<typename id_t>
105 bool IndexGenerator<id_t>::isAssigned(id_type id)
106 {
107  // If the latest id is valid, it is assigned by definition
108  if (m_latest >= 0 && id == m_latest) {
109  return true;
110  }
111 
112  // Ids past the highest one are not assigned
113  if (id > m_highest) {
114  return false;
115  }
116 
117  // Ids before the lowest one are not assigned
118  if (id < m_lowest) {
119  return false;
120  }
121 
122  // The id is assigned only if is not in the trash
123  if (m_trash.count(id) > 0) {
124  return false;
125  }
126 
127  return true;
128 }

135 template<typename id_t>
136 void IndexGenerator<id_t>::setAssigned(typename IndexGenerator<id_t>::id_type id)
137 {
138  // Check if the id has already been assigned
139  if (isAssigned(id)) {
140  throw std::runtime_error("Requested id has already been assigned.");
141  }
142 
143  // If the id is past the highest assigned id we need to trash all the ids
144  // from the highest assigned to this one (the generator only handles
145  // contiguous ids), otherwise look for the id in the trash and, if found,
146  // recycle it (if the id is not in the trash, this means it was already
147  // an assigned id).
148  if (m_highest == NULL_ID && m_lowest == NULL_ID) {
149  m_lowest = id;
150  m_highest = id;
151  } else if (id > m_highest) {
152  for (id_type wasteId = m_highest + 1; wasteId < id; ++wasteId) {
153  trash(wasteId);
154  }
155 
156  m_highest = id;
157  } else if (id < m_lowest) {
158  for (id_type wasteId = id + 1; wasteId < m_lowest; ++wasteId) {
159  trash(wasteId);
160  }
161 
162  m_lowest = id;
163  } else {
164  assert(m_trash.count(id) > 0);
165  eraseFromTrash(id);
166  }
167 
168  // This is the latest assigned id
169  m_latest = id;
170 }

179 template<typename id_t>
180 void IndexGenerator<id_t>::trash(typename IndexGenerator<id_t>::id_type id)
181 {
182  assert(m_trash.count(id) == 0);
183 
184  // If we are trashing the highest or the lowest id we can update the
185  // limits of the generator, otherwise we add the id to the trash.
186  if (id == m_highest && id == m_lowest) {
187  // If highest and lowest values are equal it means that we have
188  // generated only one id. Since we are now trashing the only id
189  // that has been generated, the trash must be emtpy.
190  assert(m_trash.empty());
191 
192  m_lowest = NULL_ID;
193  m_highest = NULL_ID;
194  } else if (id == m_highest) {
195  --m_highest;
196  while (m_trash.count(m_highest) > 0) {
197  eraseFromTrash(m_highest);
198  --m_highest;
199  }
200  } else if (id == m_lowest) {
201  ++m_lowest;
202  while (m_trash.count(m_lowest) > 0) {
203  eraseFromTrash(m_lowest);
204  ++m_lowest;
205  }
206  } else {
207  pushToTrash(id);
208  }
209 
210  // We only keep track of the latest assigned id, if we trash that id we
211  // have no information of the previous assigned id.
212  if (id == m_latest) {
213  m_latest = NULL_ID;
214  }
215 }

288 template<typename id_t>
289 void IndexGenerator<id_t>::restore(std::istream &stream)
290 {
291  // Version
292  int version;
293  utils::binary::read(stream, version);
294  if (version != getBinaryArchiveVersion()) {
295  throw std::runtime_error ("The version of the file does not match the required version");
296  }
297 
298  // Generator data
299  utils::binary::read(stream, m_latest);
300  utils::binary::read(stream, m_lowest);
301  utils::binary::read(stream, m_highest);
302 
303  size_t nTrashedIds;
304  utils::binary::read(stream, nTrashedIds);
305 
306  std::set<id_type>().swap(m_trash);
307 
308  for (size_t i = 0; i < nTrashedIds; ++i) {
309  id_type id;
310  utils::binary::read(stream, id);
311  pushToTrash(id);
312  }
313 }

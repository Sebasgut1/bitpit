174 void LevelSetCachedObject::propagateSign() {
175 
176  VolumeKernel const &mesh = *(m_kernelPtr->getMesh()) ;
177  const PiercedVector<Cell> &cells = mesh.getCells();
178 
179  VolumeKernel::CellConstIterator cellBegin = mesh.cellConstBegin();
180  VolumeKernel::CellConstIterator cellEnd = mesh.cellConstEnd();
181 
182  std::vector<long> seeds;
183  seeds.reserve(mesh.getCellCount());
184 
185  // Evaluate the bounding box of the object
186  std::array<double,3> boxMin;
187  std::array<double,3> boxMax;
188  getBoundingBox(boxMin, boxMax);
189 
190  // Set the initial propagation status of the cells
191  //
192  // We don't need to set the sign of cells associated to a levelset info,
193  // if a cell is associated to a levelset info we know its sign (the value
194  // contained in the levelset info may be a dummy value, but the sign is
195  // the correct one) and the cell can be used as seeds for propagating the
196  // sign to other cells. External sign should not be added to the seeds
197  // because we don't need to propagate the sign in the external region.
198  //
199  // The sign of the cells outise the bounding box of all objects (external
200  // cells) can be either positive or negative depending on the orientation
201  // of the objects. There is no need to propagate the sign into those cells,
202  // once the sign of the external region is know, it can be assigned to all
203  // the cells in the external region. When the propagation reaches the
204  // external region it can be stopped, the sign of the seed from which the
205  // propagation has started will be the sign of the external region.
206  PiercedStorage<int, long> propagationStatus(1, &cells);
207 
208  int externalSign = PROPAGATION_SIGN_UNDEFINED;
209  long nExternal = 0;
210  long nWaiting = mesh.getCellCount();
211  for (auto itr = cellBegin; itr != cellEnd; ++itr) {
212  long cellId = itr.getId();
213  long cellRawId = itr.getRawIndex();
214  int &cellPropagationStatus = propagationStatus.rawAt(cellRawId);
215 
216  int cellSign;
217  if (m_ls.find(cellId) != m_ls.end()) {
218  cellSign = getSign(cellId);
219  } else {
220  cellSign = PROPAGATION_SIGN_UNDEFINED;
221  }
222 
223  initializeCellSignPropagation(cellId, cellSign, boxMin, boxMax,
224  &cellPropagationStatus, &seeds,
225  &nWaiting, &nExternal, &externalSign);
226  }
227 
228  // If there are no external cells, set the sign of the external region
229  // to a dummy values. In this way the routine that propagates the sign
230  // will not try to detect the sign of the external region.
231  if (nExternal == 0) {
232  externalSign = PROPAGATION_SIGN_DUMMY;
233  }
234 
235  // Use the seeds to propagate the sign
236  propagateSeedSign(seeds, &propagationStatus, &nWaiting, &externalSign);
237 
238 #if BITPIT_ENABLE_MPI
239  // If there are cells with an unknown sign, data communication among
240  // ghost cells is needed. However it is only possibly to have cells with
241  // an unknown sign for partinioned patches.
242  long nGlobalWaiting = nWaiting;
243  if (mesh.isPartitioned()) {
244  MPI_Allreduce(MPI_IN_PLACE, &nGlobalWaiting, 1, MPI_LONG, MPI_SUM, m_kernelPtr->getCommunicator());
245  }
246 
247  if (nGlobalWaiting != 0) {
248  assert(mesh.isPartitioned());
249  assert(mesh.getProcessorCount() != 1);
250 
251  // Initialize the communicator for exchanging the sign of the ghosts
252  DataCommunicator dataCommunicator(m_kernelPtr->getCommunicator());
253 
254  int sign;
255  size_t dataSize = sizeof(sign);
256 
257  // Set the receives
258  for (const auto &entry : mesh.getGhostCellExchangeTargets()) {
259  const int rank = entry.first;
260  const auto &list = entry.second;
261 
262  dataCommunicator.setRecv(rank, list.size() * dataSize);
263  }
264 
265  // Set the sends
266  for (const auto &entry : mesh.getGhostCellExchangeSources()) {
267  const int rank = entry.first;
268  auto &list = entry.second;
269 
270  dataCommunicator.setSend(rank, list.size() * dataSize);
271  }
272 
273  // Communicate the sign among the partitions
274  while (nGlobalWaiting != 0) {
275  // Start the receives
276  for (const auto &entry : mesh.getGhostCellExchangeTargets()) {
277  const int rank = entry.first;
278  dataCommunicator.startRecv(rank);
279  }
280 
281  // Start the sends
282  for (const auto &entry : mesh.getGhostCellExchangeSources()) {
283  const int rank = entry.first;
284  const auto &sendIds = entry.second;
285  SendBuffer &buffer = dataCommunicator.getSendBuffer(rank);
286 
287  for (long cellId : sendIds) {
288  int sign = PROPAGATION_SIGN_UNDEFINED;
289  if (propagationStatus.at(cellId) == PROPAGATION_STATUS_REACHED) {
290  sign = getSign(cellId);
291  }
292  buffer << sign;
293  }
294 
295  dataCommunicator.startSend(rank);
296  }
297 
298  // Receive the sign and propagate the sign
299  //
300  // If we discover the sign of a ghost, we can use it as a seed.
301  seeds.clear();
302  int nCompletedRecvs = 0;
303  while (nCompletedRecvs < dataCommunicator.getRecvCount()) {
304  int rank = dataCommunicator.waitAnyRecv();
305  const auto &recvIds = mesh.getGhostCellExchangeTargets(rank);
306  RecvBuffer &buffer = dataCommunicator.getRecvBuffer(rank);
307 
308  // Receive data and detect new seeds
309  for (long cellId : recvIds) {
310  buffer >> sign;
311  if (sign == PROPAGATION_SIGN_UNDEFINED) {
312  continue;
313  }
314 
315  std::size_t cellRawId = cells.getRawIndex(cellId);
316  int &cellPropagationStatus = propagationStatus.rawAt(cellRawId);
317  if (cellPropagationStatus == PROPAGATION_STATUS_WAITING) {
318  // Set the sign
319  setSign(cellId, sign);
320 
321  // Initialize sign propagation
322  initializeCellSignPropagation(cellId, sign, boxMin, boxMax,
323  &cellPropagationStatus, &seeds,
324  &nWaiting, &nExternal, &externalSign);
325  } else if (cellPropagationStatus == PROPAGATION_STATUS_REACHED) {
326  assert(getSign(cellId) == sign);
327  }
328  }
329 
330  ++nCompletedRecvs;
331  }
332 
333  if (seeds.size() > 0) {
334  propagateSeedSign(seeds, &propagationStatus, &nWaiting, &externalSign);
335  }
336 
337  // Wait to the sends to finish
338  dataCommunicator.waitAllSends();
339 
340  // Update the global counter for cells with an unknow sign
341  nGlobalWaiting = nWaiting;
342  MPI_Allreduce(MPI_IN_PLACE, &nGlobalWaiting, 1, MPI_LONG, MPI_SUM, m_kernelPtr->getCommunicator());
343  }
344  }
345 
346  // Communicate the sign of the external region
347  //
348  // The sign has to be consistent among all the partitions.
349  bool exchangeExternalSign;
350  if (mesh.isPartitioned()) {
351  exchangeExternalSign = (nExternal != 0);
352  MPI_Allreduce(MPI_IN_PLACE, &exchangeExternalSign, 1, MPI_C_BOOL, MPI_LOR, m_kernelPtr->getCommunicator());
353  } else {
354  exchangeExternalSign = false;
355  }
356 
357  if (exchangeExternalSign) {
358  bool positiveExternalSign = (externalSign == 1);
359  MPI_Allreduce(MPI_IN_PLACE, &positiveExternalSign, 1, MPI_C_BOOL, MPI_LOR, m_kernelPtr->getCommunicator());
360 
361  bool negativeExternalSign = (externalSign == -1);
362  MPI_Allreduce(MPI_IN_PLACE, &negativeExternalSign, 1, MPI_C_BOOL, MPI_LOR, m_kernelPtr->getCommunicator());
363 
364  if (positiveExternalSign && negativeExternalSign) {
365  externalSign = PROPAGATION_SIGN_UNDEFINED;
366  } else if (positiveExternalSign) {
367  externalSign = 1;
368  } else if (negativeExternalSign) {
369  externalSign = -1;
370  } else {
371  externalSign = PROPAGATION_SIGN_UNDEFINED;
372  }
373  }
374 #else
375  // Check that the sign has been propagated into all regions
376  assert(nWaiting == 0);
377 #endif
378 
379  // Check that the sign of the external region was correctly identified
380  if (nExternal != 0 && externalSign == PROPAGATION_SIGN_UNDEFINED) {
381  throw std::runtime_error("Sign of external region not properly identified!");
382  }
383 
384  // Assign the sign to the external cells
385  for (auto itr = cellBegin; itr != cellEnd; ++itr) {
386  std::size_t cellRawId = itr.getRawIndex();
387  if (propagationStatus.rawAt(cellRawId) != PROPAGATION_STATUS_EXTERNAL) {
388  continue;
389  }
390 
391  long cellId = itr.getId();
392  setSign(cellId, externalSign);
393  --nExternal;
394  }
395 
396  assert(nExternal == 0);
397 }

419 void LevelSetCachedObject::initializeCellSignPropagation(long cellId, int cellSign,
420  const std::array<double, 3> &boxMin,
421  const std::array<double, 3> &boxMax,
422  int *cellStatus, std::vector<long> *seeds,
423  long *nWaiting, long *nExternal,
424  int *externalSign) {
425 
426  // Detect if the cell is external
427  //
428  // A cell is external if is outside the object bounding box.
429  const VolumeKernel &mesh = *(m_kernelPtr->getMesh());
430  double geometricTolerance = mesh.getTol();
431  std::array<double, 3> centroid = mesh.evalCellCentroid(cellId);
432 
433  bool isExternal = false;
434  for (int i = 0; i < 3; ++i) {
435  if (centroid[i] < boxMin[i] - geometricTolerance || centroid[i] > boxMax[i] + geometricTolerance) {
436  isExternal = true;
437  break;
438  }
439  }
440 
441  // Detect the status of the cell
442  //
443  // We can have three different cases:
444  // - the sign of the cell has been defined, in this case the propgation
445  // status is set as "REACHED";
446  // - there is no sign for the cell and the cell is external, in this case
447  // the propgation status is set as "EXTERNAL";
448  // - there is no sign for the cell and the cell is not external, in this
449  // case the propgation status is set as "WAITING".
450  //
451  // When the propgation reaches a cell it's possible to update the list of
452  // seeds or the external sign.
453  if (cellSign != PROPAGATION_SIGN_UNDEFINED) {
454  *cellStatus = PROPAGATION_STATUS_REACHED;
455  --(*nWaiting);
456 
457  // Non-external cells are added to the list of seeds, for external
458  // cells the external sign is updated.
459  if (!isExternal) {
460  seeds->push_back(cellId);
461  } else if (*externalSign != PROPAGATION_SIGN_DUMMY) {
462  if (cellSign != PROPAGATION_SIGN_UNDEFINED) {
463  if (*externalSign == PROPAGATION_SIGN_UNDEFINED) {
464  *externalSign = cellSign;
465  } else if (*externalSign != cellSign) {
466  throw std::runtime_error("Mismatch in sign of external region!");
467  }
468  }
469  }
470  } else {
471  if (!isExternal) {
472  *cellStatus = PROPAGATION_STATUS_WAITING;
473  } else {
474  *cellStatus = PROPAGATION_STATUS_EXTERNAL;
475  ++(*nExternal);
476  --(*nWaiting);
477  }
478  }
479 }

502 void LevelSetCachedObject::propagateSeedSign(const std::vector<long> &seeds,
503  PiercedStorage<int, long> *statuses,
504  long *nWaiting, int *externalSign) {
505 
506  VolumeKernel const &mesh = *(m_kernelPtr->getMesh()) ;
507 
508  std::vector<long> processList;
509 
510  std::size_t seedCursor = seeds.size();
511  while (seedCursor != 0) {
512  // Get a seed
513  --seedCursor;
514  long seedId = seeds[seedCursor];
515 
516  // Get the sign of the seed
517  int seedSign = getSign(seedId);
518  if (seedSign == PROPAGATION_SIGN_UNDEFINED) {
519  continue;
520  }
521 
522  // Initialize the process list with the seed
523  processList.resize(1);
524  processList[0] = seedId;
525 
526  // Propagate the sign
527  while (!processList.empty()) {
528  long cellId = processList.back();
529  processList.resize(processList.size() - 1);
530 
531  // Process non-seed cells
532  bool isSeed = (cellId == seedId);
533  if (!isSeed) {
534  // Consider only cells waiting for the propagation to reach them
535  //
536  // The process list is not unique, it can contain duplicate, so
537  // we need to make sure not to process a cell multiple times.
538  int &cellStatus = statuses->at(cellId);
539  if (cellStatus != PROPAGATION_STATUS_WAITING) {
540  continue;
541  }
542 
543  // Set the sign of the cell
544  setSign(cellId, seedSign);
545  cellStatus = PROPAGATION_STATUS_REACHED;
546  --(*nWaiting);
547 
548  // If there are no more waiting cells and we have detected the
549  // sign of the external region we can exit
550  if (*nWaiting == 0 && *externalSign != PROPAGATION_SIGN_UNDEFINED) {
551  return;
552  }
553  }
554 
555  // Process cell neighbours
556  //
557  // If a neighbour is waiting for the propagation, add it to the
558  // process list. When the propagation reaches an external cell
559  // the sign of the seed frow which the propagation started will
560  // be the sign of the external region.
561  const Cell &cell = mesh.getCell(cellId);
562  const long *cellNeighs = cell.getAdjacencies() ;
563  int nCellNeighs = cell.getAdjacencyCount() ;
564  for(int n = 0; n < nCellNeighs; ++n){
565  long neighId = cellNeighs[n] ;
566  int neighStatus = statuses->at(neighId);
567  if (neighStatus == PROPAGATION_STATUS_WAITING) {
568  processList.push_back(neighId);
569  } else if (neighStatus == PROPAGATION_STATUS_EXTERNAL) {
570  // If the sign of the external region is unknown it can
571  // be assigned, otherwise check if the current sign is
572  // consistent with the previously evaluated sign.
573  if (*externalSign == PROPAGATION_SIGN_UNDEFINED) {
574  *externalSign = seedSign;
575  } else if (*externalSign != seedSign) {
576  throw std::runtime_error("Mismatch in sign of external region!");
577  }
578  }
579  }
580  }
581  }
582 }

590 void LevelSetCachedObject::setSign(long id, int sign) {
591 
592  // The sign has to be set only if it is different from the default sign
593  if (sign == levelSetDefaults::SIGN) {
594  return;
595  }
596 
597  // Get the info associated to the id
598  PiercedVector<LevelSetInfo>::iterator infoItr = m_ls.find(id);
599  if (infoItr == m_ls.end()){
600  infoItr = m_ls.emplace(id);
601  }
602 
603  // Update the sign
604  (*infoItr).value = sign * levelSetDefaults::VALUE;
605 }

282 #if BITPIT_ENABLE_MPI
288 void PODKernel::initializeCommunicator(MPI_Comm communicator)
289 {
290  // Communication can be set just once
291  if (isCommunicatorSet())
292  throw std::runtime_error ("PODKernel communicator can be set just once");
293 
294  // The communicator has to be valid
295  if (communicator == MPI_COMM_NULL)
296  throw std::runtime_error ("PODKernel communicator is not valid");
297 
298  // Create a copy of the user-specified communicator
299  //
300  // No library routine should use MPI_COMM_WORLD as the communicator;
301  // instead, a duplicate of a user-specified communicator should always
302  // be used.
303  MPI_Comm_dup(communicator, &m_communicator);
304 }

52 void extractWithoutReplacement(int n, int m, std::vector<int> &list)
53 {
54  // Initialize variables
55  if (n > m+1) {
56  std::cout << "error" << std::endl;
57  return;
58  }
59 
60  // Resize input variables
61  list.resize(n);
62 
63  // Initialize extraction set
64  int N = m;
65  std::vector<int> set(m+1, -1);
66  for (int i = 0; i < m+1; i++) {
67  set[i] = i;
68  }
69 
70  // Extract integers without replacement
71  for (int i = 0; i < n; i++) {
72  int index = (int) round(((double) N) * ((double) rand())/((double) RAND_MAX));
73  list[i] = set[index];
74  set[index] = set[N];
75  N--;
76  }
77 }

764 template<typename value_t, typename id_t>
765 void PiercedVector<value_t, id_t>::swap(PiercedVector &x) noexcept
766 {
767  // The swap will swap also the slave-master information. This is not what
768  // we want, therefore the two pierced storage will be unregistered and the
769  // registered again after the swap. When the kernel is unset the storage
770  // can't be clear, otherwise its contents will be lost.
771  PiercedVectorStorage<value_t, id_t>::detachKernel();
772  x.PiercedVectorStorage<value_t, id_t>::detachKernel();
773 
774  // Swap kernel data
775  PiercedVectorKernel<id_t>::swap(x);
776 
777  // Swap storage data
778  PiercedVectorStorage<value_t, id_t>::swap(x);
779 
780  // Re-register the storages
781  //
782  // The function that sets a dynamic kernel may throw an exception if the
783  // kernel is already set or if we are trying to set a null kernel. Here
784  // neither of the two cases can happen, because the kernel has been
785  // previously cleared and the kernel we are trying to set is not null.
786  try {
787  PiercedVectorStorage<value_t, id_t>::setDynamicKernel(this, PiercedVectorKernel<id_t>::SYNC_MODE_DISABLED);
788  x.PiercedVectorStorage<value_t, id_t>::setDynamicKernel(&x, PiercedVectorKernel<id_t>::SYNC_MODE_DISABLED);
789  } catch (const std::runtime_error &exception) {
790  assert(false && "Error while swapping the PiercedVector!");
791  }
792 }

882 template<typename value_t, typename id_t>
883 typename PiercedVector<value_t, id_t>::iterator PiercedVector<value_t, id_t>::reclaimValue(const FillAction &action)
884 {
885  std::size_t pos = action.info[PiercedSyncAction::INFO_POS];
886  switch (static_cast<typename FillAction::FillActionType>(action.type)) {
887 
888  case FillAction::TYPE_OVERWRITE:
889  {
890  // Nothing to do
891  break;
892  }
893 
894  case FillAction::TYPE_INSERT:
895  {
896  // Since we are increasing the sotrage by an element at the time
897  // calling a reserve will hurt performance badly because this will
898  // prevent the automatic reallocation of the storage.
899  PiercedVectorStorage<value_t, id_t>::rawEmplace(pos);
900  break;
901  }
902 
903  case FillAction::TYPE_APPEND:
904  {
905  // Since we are increasing the sotrage by an element at the time
906  // calling a reserve will hurt performance badly because this will
907  // prevent the automatic reallocation of the storage.
908  PiercedVectorStorage<value_t, id_t>::rawEmplaceBack();
909  pos = PiercedVectorKernel<id_t>::getLastUsedPos();
910  break;
911  }
912 
913  default:
914  {
915  BITPIT_UNREACHABLE("This action is not handled");
916  break;
917  }
918 
919  }
920 
921  // Return the iterator to the position where the element was inserted
922  return PiercedVectorStorage<value_t, id_t>::rawFind(pos);
923 }

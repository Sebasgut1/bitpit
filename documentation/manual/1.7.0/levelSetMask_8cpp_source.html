61 LevelSetMask::LevelSetMask(int id, const std::unordered_set<long> &mask, const VolumeKernel &mesh ) :LevelSetSegmentation(id) {
62 
63  std::unordered_map<long,long> meshToEnvelope ;
64 
65  std::unique_ptr<SurfUnstructured> segmentation = extractCellEnvelope(mask,mesh,meshToEnvelope) ;
66 
67  long intrIndex = meshToEnvelope.begin()->first;
68  long enveIndex = meshToEnvelope.begin()->second;

122 std::unique_ptr<SurfUnstructured> LevelSetMask::extractCellEnvelope(const std::unordered_set<long> &mask, const VolumeKernel &mesh, std::unordered_map<long,long> &meshToEnvelope){
123 
124  std::vector<long> list;
125 
126  for( long cellIndex : mask){
127  const auto &cell = mesh.getCell(cellIndex);
128  const long *adjacencies = cell.getAdjacencies();
129  const long *interfaceIndex = cell.getInterfaces();
130 
131  int interfaceCount= cell.getInterfaceCount() ;
132 
133  for(int i=0; i<interfaceCount; i++){
134  long neigh = adjacencies[i];
135 
136  if(mask.count(neigh)==0){
137  list.push_back(interfaceIndex[i]);
138  }
139  }
140  }
141 
142  return extractFaceEnvelope(list,mesh,meshToEnvelope);
143 }

153 std::unique_ptr<SurfUnstructured> LevelSetMask::extractFaceEnvelope(const std::vector<long> &list, const VolumeKernel &mesh, std::unordered_map<long,long> &meshToEnvelope){
154 
155  std::unique_ptr<SurfUnstructured> envelope;
156 #if BITPIT_ENABLE_MPI==1
157  envelope = std::unique_ptr<SurfUnstructured>(new SurfUnstructured(mesh.getDimension()-1,mesh.getDimension(),mesh.getCommunicator()));
158 #else
159  envelope = std::unique_ptr<SurfUnstructured>(new SurfUnstructured(mesh.getDimension()-1,mesh.getDimension()));
160 #endif
161 
162 
163  // ====================================================================== //
164  // RESIZE DATA STRUCTURES //
165  // ====================================================================== //
166  long nVertices(0);
167  long nCells(list.size());
168 
169  for( long faceIndex : list){
170  auto const &interface = mesh.getInterface(faceIndex);
171  nVertices += interface.getVertexCount() ;
172  }
173 
174  envelope->reserveVertices(nVertices);
175  envelope->reserveCells(nCells);
176 
177  // ====================================================================== //
178  // LOOP OVER CELLS //
179  // ====================================================================== //
180  std::unordered_map<long,long> vertexMap;
181 
182  for( long faceIndex : list){
183  auto const &interface = mesh.getInterface(faceIndex);
184  ConstProxyVector<long> faceVertexIds = interface.getVertexIds();
185  int nFaceVertices = faceVertexIds.size();
186 
187  // Add face vertices to the envelope and get face
188  // connectivity in the envelope
189  std::unique_ptr<long[]> faceEnvelopeConnect = std::unique_ptr<long[]>(new long[nFaceVertices]);
190  for (int j = 0; j < nFaceVertices; ++j) {
191  long vertexId = faceVertexIds[j];
192 
193  // If the vertex is not yet in the envelope
194  // add it.
195  if (vertexMap.count(vertexId) == 0) {
196  const Vertex &vertex = mesh.getVertex(vertexId);
197  auto envelopeVertex = envelope->addVertex(vertex);
198  vertexMap[vertexId] = envelopeVertex->getId();
199  }
200 
201  // Update face connectivity in the envelope
202  faceEnvelopeConnect[j] = vertexMap.at(vertexId);
203  }
204 
205  // Add face to envelope
206  ElementType faceType = interface.getType();
207  PatchKernel::CellIterator cellItr = envelope->addCell(faceType, std::move(faceEnvelopeConnect));
208  meshToEnvelope.insert({{faceIndex,cellItr->getId()}});
209  }
210 
211  envelope->squeeze();
212  envelope->initializeAdjacencies();
213  envelope->initializeInterfaces();
214 
215  envelope->getVTK().setName("geometry_002") ;
216  envelope->write() ;
217 
218  return envelope;
219 
220 }

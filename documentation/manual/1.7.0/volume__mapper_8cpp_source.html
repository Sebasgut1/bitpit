152 #if BITPIT_ENABLE_MPI
158 void VolumeMapper::initializeCommunicator(MPI_Comm communicator)
159 {
160  // Communication can be set just once
161  if (isCommunicatorSet()) {
162  throw std::runtime_error ("VolumeMapper communicator can be set just once");
163  }
164 
165  // The communicator has to be valid
166  if (communicator == MPI_COMM_NULL) {
167  throw std::runtime_error ("VolumeMapper communicator is not valid");
168  }
169 
170  // Create a copy of the user-specified communicator
171  //
172  // No library routine should use MPI_COMM_WORLD as the communicator;
173  // instead, a duplicate of a user-specified communicator should always
174  // be used.
175  MPI_Comm_dup(communicator, &m_communicator);
176 
177  // Initialize parallel information
178  MPI_Comm_size(m_communicator, &m_nProcs);
179  MPI_Comm_rank(m_communicator, &m_rank);
180 }

206 void VolumeMapper::freeCommunicator()

213  MPI_Finalized(&finalizedCalled);

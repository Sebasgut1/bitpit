52 void Map::initialize(uint8_t dim){
53  m_dim = dim;
54 
55  m_origin[0] = 0.;
56  m_origin[1] = 0.;
57  m_origin[2] = 0.;
58 
59  m_L = 1.0;
60 
61  if (m_dim > 0) {
62  m_nnodes = uint8_t(1)<<m_dim;
63  m_nnodesPerFace = uint8_t(1)<<(m_dim-1);
64 
65  m_maxLength = TreeConstants::MAX_LENGTH;
66  m_maxArea = uipow<uint64_t>(m_maxLength, m_dim - 1);
67  m_maxVolume = uipow<uint64_t>(m_maxLength, m_dim);
68 
69  m_maxLength_1 = 1. / double(m_maxLength);
70  m_maxArea_1 = 1. / double(m_maxArea);
71  m_maxVolume_1 = 1. / double(m_maxVolume);
72  } else {
73  m_nnodes = 0;
74  m_nnodesPerFace = 0;
75 
76  m_maxLength = 0;
77  m_maxArea = 0;
78  m_maxVolume = 0;
79 
80  m_maxLength_1 = std::numeric_limits<double>::infinity();
81  m_maxArea_1 = std::numeric_limits<double>::infinity();
82  m_maxVolume_1 = std::numeric_limits<double>::infinity();
83  }

85 };

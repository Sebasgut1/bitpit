99 bitpit::FlatVector2D<long> Cell::createNeighbourhoodStorage(bool storeNeighbourhood)
100 {

257 void Cell::_initialize(bool interior, bool initializeNeighbourhood, bool storeNeighbourhood)
258 {
259  setInterior(interior);
260 
261  // Neighbourhood
262  if (initializeNeighbourhood) {
263  // To reduce memory fragmentation, destroy both interfaces/adjacencies
264  // before resetting the interfaces/adjacencies
265  int reallocateInterfaces = (static_cast<int>(m_interfaces.getItemCount()) != getFaceCount());
266  if (reallocateInterfaces) {
267  m_interfaces.destroy();
268  }
269 
270  int reallocateAdjacencies = (static_cast<int>(m_adjacencies.getItemCount()) != getFaceCount());
271  if (reallocateAdjacencies) {
272  m_adjacencies.destroy();
273  }
274 
275  // Reset the interfaces/adjacencies
276  resetInterfaces(storeNeighbourhood);
277  resetAdjacencies(storeNeighbourhood);
278  }
279 }

811 void Cell::display(std::ostream &out, unsigned short int indent) const
812 {
813  // ====================================================================== //
814  // VARIABLES DECLARATION //
815  // ====================================================================== //
816 
817  // Local variables
818  std::string t_s(indent, ' ');
819 
820  // Counters
821  int i, j;
822  int nn;
823 
824  // ====================================================================== //
825  // DISPLAY INFOS //
826  // ====================================================================== //
827  if (getType() == ElementType::UNDEFINED) {
828  out << t_s << "cell type: (unknown)" << std::endl;
829  return;
830  }
831 
832  // Scope variables -------------------------------------------------- //
833  int nv = getVertexCount();
834  int nf = getFaceCount();
835 
836  ConstProxyVector<long> cellVertexIds = getVertexIds();
837 
838  // General info ----------------------------------------------------- //
839  out << t_s << "cell type: " << getType() << std::endl;
840  out << t_s << "ID: " << getId() << std::endl;
841  out << t_s << "is ghost: ";
842  if (m_interior) { out << "(false)"; }
843  else { out << "(true)"; }
844  out << std::endl;
845 
846  // Connectivity infos --------------------------------------------------- //
847  out << t_s << "connectivity: [ ";
848  for (i = 0; i < nv-1; ++i) {
849  if (cellVertexIds[i] == Vertex::NULL_ID) out << "n.a. ";
850  else out << cellVertexIds[i] << ", ";
851  } //next i
852  if (cellVertexIds[nv-1] == Vertex::NULL_ID) out << "n.a. ";
853  else out << cellVertexIds[nv-1] << " ]" << std::endl;
854 
855  // neighbors infos ------------------------------------------------------ //
856  if (m_adjacencies.size() > 0) {
857  out << t_s << "neighbors: [ ";
858  for (i = 0; i < nf; ++i) {
859  nn = getAdjacencyCount(i);
860  out << "[ ";
861  if (nn == 0) {
862  out << "n.a. ";
863  } else {
864  for (j = 0; j < nn; ++j) {
865  out << getAdjacency(i, j);
866  if (j != (nn - 1)) {
867  out << ",";
868  }
869  out << " ";
870  }
871  }
872  out << "]";
873  if (i != (nf - 1)) {
874  out << ",";
875  }
876  out << " ";
877  }
878  out << "]" << std::endl;
879  }
880 
881  // interface infos ------------------------------------------------------ //
882  if (m_interfaces.size() > 0) {
883  out << t_s << "interfaces: [ ";
884  for (i = 0; i < nf; ++i) {
885  nn = getInterfaceCount(i);
886  out << "[ ";
887  if (nn == 0) {
888  out << "n.a. ";
889  } else {
890  for (j = 0; j < nn; ++j) {
891  out << getInterface(i, j);
892  if (j != (nn - 1)) {
893  out << ",";
894  }
895  out << " ";
896  }
897  }
898  out << "]";
899  if (i != (nf - 1)) {
900  out << ",";
901  }
902  out << " ";
903  }
904  out << "]" << std::endl;
905  }
906 
907 }

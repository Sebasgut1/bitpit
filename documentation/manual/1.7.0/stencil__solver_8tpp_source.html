58 #else
64 template<typename stencil_t>
65 DiscretizationStencilSolverAssembler<stencil_t>::DiscretizationStencilSolverAssembler(const std::vector<stencil_t> *stencils)
66 #endif
67  : StencilSolverAssembler(), m_stencils(stencils),
68  m_blockSize(-1)
69 {
70  // Initialize block size
71  initializeBlockSize();
72 
73  // Count the DOFs
74  m_nDOFs = stencils->size();
75 
76 #if BITPIT_ENABLE_MPI==1
77  m_nGlobalDOFs = m_nDOFs;
78  if (partitioned) {
79  MPI_Allreduce(MPI_IN_PLACE, &m_nGlobalDOFs, 1, MPI_LONG, MPI_SUM, communicator);
80  }
81 #endif
82 
83  // Count maximum non-zero elements
84  m_maxRowNZ = 0;
85  for (long n = 0; n < getRowCount(); ++n) {
86  m_maxRowNZ = std::max(getRowNZCount(n), m_maxRowNZ);
87  }
88 
89 #if BITPIT_ENABLE_MPI==1
90  // Get offsets
91  m_globalDOFOffset = 0;
92  if (partitioned) {
93  int nProcessors;
94  MPI_Comm_size(communicator, &nProcessors);
95 
96  std::vector<long> nRankDOFs(nProcessors);
97  MPI_Allgather(&m_nDOFs, 1, MPI_LONG, nRankDOFs.data(), 1, MPI_LONG, communicator);
98 
99  int rank;
100  MPI_Comm_rank(communicator, &rank);
101  for (int i = 0; i < rank; ++i) {
102  m_globalDOFOffset += nRankDOFs[i];
103  }
104  }
105 #endif
106 }

219 template<typename stencil_t>
220 void DiscretizationStencilSolverAssembler<stencil_t>::getRowPattern(long rowIndex, ConstProxyVector<long> *pattern) const
221 {
222  // Get stencil information
223  const stencil_t &stencil = (*m_stencils)[rowIndex];
224  std::size_t stencilSize = stencil.size();
225 
226  // Get pattern
227  const long *patternData = stencil.patternData();
228  if (m_blockSize == 1) {
229  pattern->set(patternData, stencilSize);
230  } else {
231  std::size_t expandedPatternSize = m_blockSize * stencilSize;
232  long *expandedPatternStorage = pattern->set(expandedPatternSize);
233 
234  std::size_t expandedPatternIdx = 0;
235  for (std::size_t k = 0; k < stencilSize; ++k) {
236  long patternBlockOffset = patternData[k] * m_blockSize;
237  for (int i = 0; i < m_blockSize; ++i) {
238  expandedPatternStorage[expandedPatternIdx++] = patternBlockOffset + i;
239  }
240  }
241  }
242 }

539 void DiscretizationStencilSolver<stencil_t>::update(std::size_t nRows, const long *rows, const std::vector<stencil_t> &stencils)
540 {

565 template<typename stencil_t>
566 void DiscretizationStencilSolver<stencil_t>::solve()
567 {
568  // Check if the stencil solver is assembled
569  if (!isAssembled()) {
570  throw std::runtime_error("Unable to solve the system. The stencil solver is not yet assembled.");
571  }
572 
573  // Subtract constant terms to the RHS
574  long nUnknowns = getRowCount();
575  double *raw_rhs = getRHSRawPtr();
576  for (long i = 0; i < nUnknowns; ++i) {
577  raw_rhs[i] -= m_constants[i];
578  }
579  restoreRHSRawPtr(raw_rhs);
580 
581  // Solve the system
582  SystemSolver::solve();
583 }

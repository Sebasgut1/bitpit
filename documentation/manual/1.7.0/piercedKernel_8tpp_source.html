227 template<typename id_t>
228 typename PiercedKernel<id_t>::ResizeAction PiercedKernel<id_t>::_resize(std::size_t n)
229 {
230  // If the size of the kernel is already the requested size there is
231  // nothing to do.
232  if (n == size()) {
233  ResizeAction syncAction(ResizeAction::TYPE_NOOP);
234 
235  return syncAction;
236  }
237 
238  // A request for a size equal to 0 is equivalent to a clear.
239  if (n == 0) {
240  _clear();
241 
242  ResizeAction syncAction(ResizeAction::TYPE_CLEAR);
243 
244  return syncAction;
245  }
246 
247  // If the requested size is greater that the current size we may need to
248  // reserve space to allow the kernel to reach the requested size.
249  if (n > size()) {
250  _reserve(n);
251 
252  ResizeAction syncAction(ResizeAction::TYPE_RESERVE);
253  syncAction.info[PiercedSyncAction::INFO_SIZE] = rawSize();
254 
255  return syncAction;
256  }
257 
258  // If the requested size is smaller that the current size
259  // we need to perform a real resize.
260 
261  // Flush holes
262  holesFlush();
263 
264  // Find the id of the last element
265  id_t last_stored_id = getSizeMarker(n - 1);
266 
267  // Find the last position
268  std::size_t last_used_pos = getPos(last_stored_id);
269 
270  // Shrink the kernel
271  rawShrink(last_used_pos + 1);
272 
273  // Generate the sync action
274  ResizeAction syncAction(ResizeAction::TYPE_RESIZE);
275  syncAction.info[PiercedSyncAction::INFO_SIZE] = rawSize();
276 
277  return syncAction;
278 }

356 template<typename id_t>
357 typename PiercedKernel<id_t>::SortAction PiercedKernel<id_t>::_sort(std::size_t beginPos, std::size_t endPos)
358 {
359  // Squeeze the kernel
360  //
361  // After the squeeze there will be no holes in the kernel.
362  SqueezeAction squeezeAction = _squeeze();
363  const std::vector<std::size_t> &squeezePermutations = *(squeezeAction.data);
364  bool squeezed = (squeezeAction.data.get() != nullptr);
365 
366  // Get updated kernel size
367  std::size_t updatedKernelRawSize = rawSize();
368 
369  // Update the sort range
370  if (squeezed) {
371  std::size_t previousBeginPos = beginPos;
372  std::size_t previousEndPos = endPos;
373  for (std::size_t i = m_begin_pos; i < m_end_pos; ++i) {
374  std::size_t previousPos = squeezePermutations[i];
375  if (previousPos == previousBeginPos) {
376  beginPos = i;
377  }
378  if ((previousPos + 1) == previousEndPos) {
379  endPos = i + 1;
380  break;
381  }
382  }
383  }
384 
385  // Evaluates the sort permutations
386  std::vector<std::size_t> sortPermutations(updatedKernelRawSize);
387  for (std::size_t i = 0; i < updatedKernelRawSize; ++i) {
388  sortPermutations[i] = i;
389  }
390 
391  std::sort(sortPermutations.begin() + beginPos, sortPermutations.begin() + endPos, idLess(m_ids));
392 
393  // Create the sync action
394  SortAction syncAction;
395  syncAction.info[PiercedSyncAction::INFO_SIZE] = updatedKernelRawSize;
396  if (squeezed) {
397  syncAction.importData(std::vector<std::size_t>(squeezePermutations));
398  std::vector<std::size_t> &permutations = *(syncAction.data);
399  for (std::size_t i = beginPos; i < endPos; ++i) {
400  permutations[i] = squeezePermutations[sortPermutations[i]];
401  }
402  } else {
403  syncAction.importData(std::vector<std::size_t>(sortPermutations));
404  }
405 
406  // Sort the ids
407  //
408  // NOTE: the reorder function will destroy the permutation vector
409  utils::reorderVector<id_t>(sortPermutations, m_ids, updatedKernelRawSize);
410 
411  // Update the positions
412  m_pos.clear();
413  for (std::size_t i = 0; i < updatedKernelRawSize; ++i) {
414  m_pos[m_ids[i]] = i;
415  }
416 
417  // Return the permutations
418  return syncAction;
419 }

454 template<typename id_t>
455 typename PiercedKernel<id_t>::SqueezeAction PiercedKernel<id_t>::_squeeze()
456 {
457  // Flush changes
458  flush();
459 
460  // Sort the holes
461  //
462  // After being flushed, the container will contain only regular holes.
463  holesSortRegular();
464 
465  // Get kernel size
466  std::size_t kernelSize = size();
467  std::size_t kernelRawSize = rawSize();
468 
469  // Initialize the sync action
470  SqueezeAction syncAction;
471  syncAction.info[PiercedSyncAction::INFO_SIZE] = kernelSize;
472 
473  // Compact the kernel
474  std::size_t nHoles = holesCount();
475  if (nHoles != 0) {
476  // Initialize sync data
477  syncAction.importData(std::vector<std::size_t>(kernelRawSize));
478  std::vector<std::size_t> &permutations = *(syncAction.data);
479 
480  // Move the elements
481  std::size_t firstPosToUpdate;
482  if (m_begin_pos == 0) {
483  firstPosToUpdate = m_holes[m_holes_regular_end - 1];
484  } else {
485  firstPosToUpdate = 0;
486  }
487 
488  for (std::size_t pos = 0; pos < firstPosToUpdate; ++pos) {
489  permutations[pos] = pos;
490  }
491 
492  std::size_t offset = 0;
493  for (std::size_t pos = firstPosToUpdate; pos < m_end_pos; pos++) {
494  if (offset < nHoles && m_holes[m_holes_regular_end - offset - 1] == pos) {
495  permutations[m_end_pos - 1 - offset] = pos;
496  ++offset;
497  continue;
498  }
499 
500  id_t id = m_ids[pos];
501  std::size_t updatedPos = pos - offset;
502 
503  setPosId(updatedPos, id);
504  setPosEmptyId(pos, pos + 1);
505  permutations[updatedPos] = pos;
506  }
507 
508  for (std::size_t pos = m_end_pos; pos < rawSize(); ++pos) {
509  permutations[pos] = pos;
510  }
511 
512  // Clear the holes
513  holesClear(true);
514 
515  // Reset begin position
516  setBeginPos(0);
517 
518  // Shrink the kernel
519  rawShrink(kernelSize);
520  }
521 
522  // Shrink to fit
523  _shrinkToFit();
524 
525  return syncAction;
526 }

853 template<typename id_t>

855 {
856  std::size_t pos = getPos(id);
857 
858  // Initialize flat id with the position of the element
859  std::size_t flat = pos;
860 
861  // Subtract pending holes before position
862  if (holesCountPending() > 0) {
863  holes_const_iterator pending_begin_itr = m_holes.cbegin() + m_holes_pending_begin;
864  holes_const_iterator pending_end_itr = m_holes.cbegin() + m_holes_pending_end;
865 
866  std::size_t nHolesBefore = 0;
867  for (auto itr = pending_begin_itr; itr != pending_end_itr; ++itr) {
868  if (*itr <= pos) {
869  ++nHolesBefore;
870  }
871  }
872 
873  flat -= nHolesBefore;
874  }
875 
876  // Subtract regular holes before position
877  if (holesCountRegular() > 0) {
878  holes_const_iterator regular_begin_itr = m_holes.cbegin() + m_holes_regular_begin;
879  holes_const_iterator regular_end_itr = m_holes.cbegin() + m_holes_regular_end;
880 
881  std::size_t nHolesBefore = 0;
882  for (auto itr = regular_begin_itr; itr != regular_end_itr; ++itr) {
883  if (*itr <= pos) {
884  ++nHolesBefore;
885  }
886  }
887 
888  flat -= nHolesBefore;
889  }
890 
891  // Done
892  return flat;
893 }

950 template<typename id_t>
951 id_t PiercedKernel<id_t>::getSizeMarker(std::size_t targetSize, const id_t &fallback)
952 {
953  // If the size is zero, we return the first element, if the target
954  // size is equal to the size minus one we return the last element,
955  // if the target size is greater or equal the current kernel size
956  // we return the fallback value.
957  if (targetSize >= size()) {
958  return fallback;
959  } else if (targetSize == 0) {
960  return m_ids[m_begin_pos];
961  } else if (targetSize == (size() - 1)) {
962  return m_ids[m_end_pos - 1];
963  }
964 
965  // Sort the holes
966  holesSortRegular();
967  holesSortPending();
968 
969  // Iterate to find the position before wihch there is the
970  // requeste number of element.
971  holes_iterator regular_begin_itr = m_holes.begin() + m_holes_regular_begin;
972  holes_iterator regular_hole_itr = m_holes.begin() + m_holes_regular_end;
973 
974  holes_iterator pending_begin_itr = m_holes.begin() + m_holes_pending_begin;
975  holes_iterator pending_hole_itr = m_holes.begin() + m_holes_pending_end;
976 
977  std::size_t nEmpties = 0;
978  std::size_t markerPos = targetSize;
979  while (true) {
980  if (isPosEmpty(markerPos)) {
981  markerPos = findNextUsedPos(markerPos - 1);
982  }
983 
984  // Count the number of holes and pending deletes before the
985  // current marker position
986  if (regular_hole_itr != regular_begin_itr) {
987  holes_iterator itr_previous = regular_hole_itr;
988  regular_hole_itr = std::upper_bound(regular_begin_itr, regular_hole_itr, markerPos, std::greater<std::size_t>());
989  nEmpties += std::distance(regular_hole_itr, itr_previous);
990  }
991 
992  if (pending_hole_itr != pending_begin_itr) {
993  holes_iterator itr_previous = pending_hole_itr;
994  pending_hole_itr = std::upper_bound(pending_begin_itr, pending_hole_itr, markerPos, std::greater<std::size_t>());
995  nEmpties += std::distance(pending_hole_itr, itr_previous);
996  }
997 
998  // Get the marker size
999  //

1341 template<typename id_t>
1342 typename PiercedKernel<id_t>::FillAction PiercedKernel<id_t>::fillHole(std::size_t hole, id_t id)
1343 {
1344  // Validate the id
1345  validateId(id);
1346 
1347  // Get the position of the hole
1348  std::size_t pos = m_holes[hole];
1349 
1350  // Fill the hole
1351  setPosId(pos, id);
1352 
1353  // Remove the holes from the kernel
1354  if (hole >= m_holes_pending_begin) {
1355  int nPendings = holesCountPending();
1356  if (nPendings > 1) {
1357  if (hole == (m_holes_pending_end - 1)) {
1358  --m_holes_pending_end;
1359  } else if (hole == m_holes_pending_begin) {
1360  ++m_holes_pending_begin;
1361  } else {

1363  }
1364  } else {
1365  holesClearPending();
1366  }
1367  } else {
1368  int nRegulars = holesCountRegular();
1369  if (nRegulars > 1) {

1372  } else if (hole == m_holes_regular_begin) {
1373  ++m_holes_regular_begin;
1374  } else {

1376  }
1377  } else {
1378  holesClearRegular();
1379  }
1380  }

1383  //
1384  // There are no holes past the end of the vector, this means that only
1385  // the begin position may need an update.
1386  if (pos < m_begin_pos) {
1387  setBeginPos(pos);
1388  }
1389

1393  // current one are correctly set, otherwise it will not be possible

1408  }

1912 void PiercedKernel<id_t>::holesFlush()
1913 {

1915  if (m_holes_pending_begin == m_holes_pending_end) {
1916  return;
1917  }
1918 
1919  // Update the id of the empty elements
1920  //
1921  // The list of pending holes is sorted, in this way we can iterate
1922  // from the last pending hole in the kernel to the first one.
1923  // We start updating the id of the last pending hole, updating also
1924  // the id of the contiguous holes before that one. Then we advance
1925  // to the next hole, skipping the positions that have already been
1926  // updated.
1927  holesSortPending();
1928

1930  std::size_t pos = m_end_pos;

1933  hole++;
1934  continue;
1935  }
1936

1938  std::size_t next_used_pos = findNextUsedPos(pos);
1939  do {
1940  setPosEmptyId(pos, next_used_pos);
1941  if (pos > 0) {
1942  pos--;
1943  } else {
1944  break;
1945  }
1946  } while (isPosEmpty(pos));
1947  } while (pos > 0 && hole != m_holes_pending_end);
1948

1950  for (std::size_t hole = m_holes_pending_begin; hole < m_holes_pending_end; ++hole) {
1951  std::size_t pos = m_holes[hole];

1955  if (m_holes_regular_begin != 0) {

1961  m_holes_regular_sorted = false;
1962  }
1963  } else {

1965  m_holes[m_holes_regular_end] = pos;
1966  }
1967  ++m_holes_regular_end;
1968  ++m_holes_pending_begin;
1969

1976  }

1989  }

1992  holesClearPending();
1993 
1994  // There are no more dirty positions
1995  m_dirty_begin_pos = m_end_pos;

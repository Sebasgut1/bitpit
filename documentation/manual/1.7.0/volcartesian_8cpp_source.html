272 void VolCartesian::_updateInterfaces()
273 {
274  // Partial updates are not supported
275  bool partialUpdate = false;
276  for (auto cellIterator = cellBegin(); cellIterator != cellEnd(); ++cellIterator) {
277  long cellId = cellIterator.getId();
278  if (!testCellAlterationFlags(cellId, FLAG_INTERFACES_DIRTY)) {
279  partialUpdate = true;
280  break;
281  }
282  }
283 
284  if (partialUpdate) {
285  log::cout() << " It is not possible to partially update the interfaces.";
286  log::cout() << " All interface will be updated.";
287  }
288 
289  // Count the total number of interfaces
290  m_nInterfaces = 0;
291  for (int n = 0; n < getDimension(); n++) {
292  std::array<int, 3> interfaceCount1D = getInterfaceCountDirection(n);
293 
294  int nDirectionInterfaces = 1;
295  for (int n = 0; n < getDimension(); n++) {
296  nDirectionInterfaces *= interfaceCount1D[n];
297  }
298  m_nInterfaces += nDirectionInterfaces;
299  }
300 
301  // Build interfaces
302  if (getMemoryMode() == MemoryMode::MEMORY_NORMAL) {
303  // Enable advanced editing
304  setExpert(true);
305 
306  // Update interfaces
307  addInterfaces();
308 
309  // Disable advanced editing
310  setExpert(false);
311  } else {
312  // Set interfaces build strategy
313  setInterfacesBuildStrategy(INTERFACES_AUTOMATIC);
314  }
315 }

419 void VolCartesian::setDiscretization(const std::array<int, 3> &nCells)
420 {
421  // Spacing
422  for (int d = 0; d < getDimension(); ++d) {
423  // Initialize cells
424  if (nCells[d] > 0) {
425  m_nCells1D[d] = nCells[d];
426  m_cellSpacings[d] = (m_maxCoords[d] - m_minCoords[d]) / m_nCells1D[d];
427 
428  m_cellCenters[d].resize(m_nCells1D[d]);
429  for (int i = 0; i < m_nCells1D[d]; i++) {
430  m_cellCenters[d][i] = m_minCoords[d] + (0.5 + i) * m_cellSpacings[d];
431  }
432  } else {
433  m_nCells1D[d] = 0;
434  m_cellSpacings[d] = 0.;
435  }
436 
437  log::cout() << " - Cell count along direction " << d << " : " << m_nCells1D[d] << "\n";
438 
439  // Initialize vertices
440  m_nVertices1D[d] = m_nCells1D[d] + 1;
441 
442  m_vertexCoords[d].resize(m_nVertices1D[d]);
443  for (int i = 0; i < m_nVertices1D[d]; i++) {
444  m_vertexCoords[d][i] = m_minCoords[d] + i * m_cellSpacings[d];
445  }
446  }
447 
448  for (int d = getDimension(); d < 3; ++d) {
449  m_nCells1D[d] = 0;
450  m_cellSpacings[d] = 0.;
451 
452  m_nVertices1D[d] = 1;
453  }
454 
455  log::cout() << std::endl;
456 
457  // Define direction ordering
458  //
459  // It is the ordering in which the different directions will be considered
460  // when numbering cell, vertices, ... For each direction, it will contain
461  // the order in which the direction should be considered or -1 if there
462  // are no cells along the direction.
463  for (int d = 0; d < 3; ++d) {
464  if (d == 0) {
465  m_directionOrdering[d] = 0;
466  } else {
467  int previousDirection = m_directionOrdering[d - 1];
468  if (previousDirection < 0 || previousDirection >= 2) {
469  m_directionOrdering[d] = -1;
470  continue;
471  }
472 
473  m_directionOrdering[d] = previousDirection + 1;
474  }
475 
476  while (m_nCells1D[m_directionOrdering[d]] == 0) {
477  if (m_directionOrdering[d] == 2) {
478  m_directionOrdering[d] = -1;
479  break;
480  }
481 
482  ++m_directionOrdering[d];
483  }
484  }
485 
486  // Count the total number of vertices
487  m_nVertices = 1;
488  for (int n = 0; n < getDimension(); ++n) {
489  m_nVertices *= m_nVertices1D[n];
490  }
491  log::cout() << " - Total vertex count: " << m_nVertices << "\n";
492 
493  // Count the total number of cells
494  m_nCells = 1;
495  for (int d = 0; d < getDimension(); ++d) {
496  if (m_directionOrdering[d] >= 0) {
497  m_nCells *= m_nCells1D[d];
498  }
499  }
500 
501  log::cout() << " - Total cell count: " << m_nCells << "\n";
502 
503  // Cell volume
504  initializeCellVolume();
505 
506  // Interface area
507  initializeInterfaceArea();
508 
509  // Update patch data structures
510  if (getMemoryMode() == MemoryMode::MEMORY_NORMAL) {
511  // Switch to light mode to reset patchdata structures
512  switchMemoryMode(MemoryMode::MEMORY_LIGHT);
513 
514  // Swich back to normal mode to rebuild patchdata structures
515  switchMemoryMode(MemoryMode::MEMORY_NORMAL);
516  }
517 }

791 void VolCartesian::switchMemoryMode(MemoryMode mode)
792 {
793  if (mode == getMemoryMode()) {

813  setSpawnStatus(SPAWN_NEEDED);
814 
815  // Set the light memory mode
816  setMemoryMode(mode);
817 
818  break;
819 
820  }
821 }

868 std::vector<adaption::Info> VolCartesian::_spawn(bool trackSpawn)
869 {
870  std::vector<adaption::Info> updateInfo;
871 
872  // If the patch is in 'normal' mode there is nothing to do.
873  if (getMemoryMode() == MEMORY_NORMAL) {
874  return updateInfo;
875  }
876 
877  // Enable advanced editing
878  setExpert(true);
879 
880  // Definition of the mesh
881  addVertices();
882  addCells();
883  if (getInterfacesBuildStrategy() == INTERFACES_AUTOMATIC) {
884  addInterfaces();
885  }
886 
887  // Disable advanced editing
888  setExpert(false);
889 
890  // Adaption info
891  if (trackSpawn) {
892  updateInfo.emplace_back();
893  adaption::Info &adaptionCellInfo = updateInfo.back();
894  adaptionCellInfo.type = adaption::TYPE_CREATION;
895  adaptionCellInfo.entity = adaption::ENTITY_CELL;
896  adaptionCellInfo.current.reserve(m_cells.size());
897  for (auto &cell : m_cells) {
898  adaptionCellInfo.current.emplace_back();
899  long &cellId = adaptionCellInfo.current.back();
900  cellId = cell.getId();
901  }
902 
903  updateInfo.emplace_back();
904  adaption::Info &adaptionInterfaceInfo = updateInfo.back();
905  adaptionInterfaceInfo.type = adaption::TYPE_CREATION;
906  adaptionInterfaceInfo.entity = adaption::ENTITY_INTERFACE;
907  adaptionInterfaceInfo.current.reserve(m_interfaces.size());
908  for (auto &interface : m_interfaces) {
909  adaptionInterfaceInfo.current.emplace_back();
910  long &interfaceId = adaptionInterfaceInfo.current.back();
911  interfaceId = interface.getId();
912  }
913  } else {
914  updateInfo.emplace_back();
915  }
916 
917  // Updating the adaption brings the patch is in normal memory mode
918  setMemoryMode(MemoryMode::MEMORY_NORMAL);
919 
920  // Done
921  return updateInfo;
922 }

953 void VolCartesian::addCells()
954 {
955  log::cout() << " >> Creating cells\n";
956 
957  // Info on the cells
958  ElementType cellType = getCellType();
959 
960  // Create the cells
961  log::cout() << " - Cell count: " << m_nCells << "\n";
962 
963  m_cells.reserve(m_nCells);
964  for (int k = 0; (isThreeDimensional()) ? (k < m_nCells1D[Vertex::COORD_Z]) : (k <= 0); k++) {
965  for (int j = 0; j < m_nCells1D[Vertex::COORD_Y]; j++) {
966  for (int i = 0; i < m_nCells1D[Vertex::COORD_X]; i++) {
967  long id_cell = getCellLinearId(i, j, k);
968  CellIterator cellIterator = VolumeKernel::addCell(cellType, id_cell);
969  Cell &cell = *cellIterator;
970 
971  // Connettività
972  long *cellConnect = cell.getConnect();
973 
974  cellConnect[0] = getVertexLinearId(i, j, k);
975  cellConnect[1] = getVertexLinearId(i + 1, j, k);
976  cellConnect[2] = getVertexLinearId(i, j + 1, k);
977  cellConnect[3] = getVertexLinearId(i + 1, j + 1, k);
978  if (cellType == ElementType::VOXEL) {
979  cellConnect[4] = getVertexLinearId(i, j, k + 1);

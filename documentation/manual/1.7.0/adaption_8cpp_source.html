135 std::size_t InfoCollection::create(Type type, Entity entity, int rank)
136 {
137  infoData_t infoData = infoData_t(type, entity, rank);
138  bool useCache = (m_cachedTypes.count(type) > 0);
139 
140  if (useCache) {
141  if (m_cache.count(infoData) > 0) {;
142  return m_cache.at(infoData);
143  }
144  }
145 
146  m_collection.emplace_back(type, entity, rank);
147 
148  std::size_t id = m_collection.size() - 1;
149  if (useCache) {
150  m_cache.insert({{infoData, id}});
151  }
152  return id;
153 }

336 void CellFlatMapping::update(const std::vector<adaption::Info> &adaptionData)
337 {
338  // Previous number of cells
339  long nPreviousCells = m_numbering.size();
340 
341  // Current number of cells
342  long nCurrentCells = m_patch->getCellCount();
343 
344  // Find the first change
345  long firstChangedFlatId = std::min(nCurrentCells, nPreviousCells);
346  long firstChangedId = m_patch->getCells().getSizeMarker(firstChangedFlatId, Element::NULL_ID);
347  for (auto &adaptionInfo : adaptionData) {
348  if (adaptionInfo.entity != adaption::ENTITY_CELL) {
349  continue;
350  }
351 
352  for (const auto &currentId : adaptionInfo.current) {
353  long currentFlatId = m_patch->getCells().evalFlatIndex(currentId);
354  if (currentFlatId < firstChangedFlatId) {
355  firstChangedId = currentId;
356  firstChangedFlatId = currentFlatId;
357  }
358  }
359  }
360 
361  // The mapping, until the flat id with the first change, is a no-op mapping
362  m_mapping.resize(nCurrentCells);
363  for (long flatId = 0; flatId < firstChangedFlatId; ++flatId) {
364  m_mapping[flatId] = flatId;
365  }
366 
367  // If there are no changes the mapping is already updated
368  if (firstChangedId < 0) {
369  return;
370  }
371 
372  // List of cells that have been deleted
373  std::unordered_set<long> removedIds;
374  for (auto &adaptionInfo : adaptionData) {
375  if (adaptionInfo.entity != adaption::ENTITY_CELL) {
376  continue;
377  }
378 
379  for (const auto &previousId : adaptionInfo.previous) {
380  removedIds.insert(previousId);
381  }
382  }
383  long nRemovedCells = removedIds.size();
384 
385  // Create a mapping between cell ids and previous flat ids
386  std::unordered_map<long, long> previousFlatIds;
387  std::unordered_map<long, long> removedFlatIds;
388  previousFlatIds.reserve(nCurrentCells - firstChangedFlatId - nRemovedCells);
389  removedFlatIds.reserve(nRemovedCells);
390  for (long previousFlatId = firstChangedFlatId; previousFlatId < nPreviousCells; ++previousFlatId) {
391  long previousId = m_numbering[previousFlatId];
392  if (removedIds.count(previousId) > 0) {
393  removedFlatIds.insert({{previousId, previousFlatId}});
394  } else {
395  previousFlatIds.insert({{previousId, previousFlatId}});
396  }
397  }
398 
399  // Add to the mapping the added cells
400  for (auto &adaptionInfo : adaptionData) {
401  if (adaptionInfo.entity != adaption::ENTITY_CELL) {
402  continue;
403  }
404 
405  // Ancestor flat index of the added cells
406  long ancestorFlatId;
407  if (adaptionInfo.previous.size() > 0) {
408  ancestorFlatId = removedFlatIds.at(adaptionInfo.previous[0]);
409  } else {
410  ancestorFlatId = -1;
411  }
412 
413  // Mapping of the added cells
414  for (const auto &currentId : adaptionInfo.current) {
415  previousFlatIds.insert({{currentId, ancestorFlatId}});
416  }
417  }
418 
419  std::unordered_map<long, long>().swap(removedFlatIds);
420 
421  // Update numbering and mapping past the flat id with the first change
422  m_numbering.resize(nCurrentCells);
423  auto cellIterator = m_patch->getCells().find(firstChangedId);
424  for (long flatId = firstChangedFlatId; flatId < nCurrentCells; ++flatId) {
425  long cellId = cellIterator->getId();
426 
427  m_numbering[flatId] = cellId;
428  m_mapping[flatId] = previousFlatIds[cellId];
429 
430  cellIterator++;
431  }
432 }
